```python
class TelecomScenarioDetector:
    """电信业务场景识别器"""

    SCENARIOS = {
        '营业厅受理': ['营业厅', '营业员', '受理', '办理'],
        '客服支撑': ['客服', '10000', '热线', '咨询'],
        '渠道管理': ['渠道', '代理商', '社会化渠道'],
        '网络保障': ['网络', '基站', '信号', '覆盖'],
        '营销活动': ['营销', '活动', '促销', '优惠']
    }

    @staticmethod
    def detect(text):
        """自动识别业务场景"""
        scores = {}
        for scenario, keywords in TelecomScenarioDetector.SCENARIOS.items():
            score = sum(1 for kw in keywords if kw in text)
            scores[scenario] = score

        return max(scores, key=scores.get) if max(scores.values()) > 0 else '通用场景'
```

```python
            'mitigation': '持续训练、人工兜底、反馈优化'
        },
        '用户体验风险': {
            'description': '用户不接受AI服务方式',
            'impact': '中',
            'probability': '中',
            'mitigation': '交互优化、培训引导、人工备选'
        },
        '性能瓶颈风险': {
            'description': '高并发场景下系统性能不足',
            'impact': '高',
            'probability': '低',
            'mitigation': '压力测试、弹性扩容、降级策略'
        }
    }

    @staticmethod
    def analyze(project_type):
        """根据项目类型返回风险清单"""
        risks = []

        if 'AI' in project_type or '智能' in project_type:
            risks.append(TelecomRiskAnalyzer.RISK_TEMPLATES['AI准确率风险'])

        if '受理' in project_type or '办理' in project_type:
            risks.extend([
                TelecomRiskAnalyzer.RISK_TEMPLATES['数据安全风险'],
                TelecomRiskAnalyzer.RISK_TEMPLATES['系统对接风险'],
                TelecomRiskAnalyzer.RISK_TEMPLATES['业务合规风险']
            ])

        return risks
```

```python
class EnhancedQualityScorer:
    """增强版质量评分器"""

    SCORING_DIMENSIONS = {
        '完整性': {
            'weight': 20,
            'checklist': [
                '需求背景完整',
                '需求描述详细',
                '验收标准明确',
                '风险识别充分'
            ]
        },
        '正确性': {
            'weight': 15,
            'checklist': [
                '业务逻辑正确',
                '技术方案可行',
                '数据流向清晰'
            ]
        },
        '专业性': {
            'weight': 20,
            'checklist': [
                '电信行业术语准确',
                '系统对接规范完整',
                '合规要求明确'
            ]
        },
        '可执行性': {
            'weight': 15,
            'checklist': [
                '需求优先级明确',
                '实施路径清晰',
                '资源需求合理'
            ]
        },
        '可验收性': {
            'weight': 15,
            'checklist': [
                '验收标准量化',
                '测试用例覆盖',
                '验收方法明确'
            ]
        },
        '可维护性': {
            'weight': 15,
            'checklist': [
                '文档结构清晰',
                '版本管理规范',
                '变更追溯完整'
            ]
        }
    }

    @staticmethod
    def score(document):
        """计算综合质量评分"""
        total_score = 0
        details = {}

        for dimension, config in EnhancedQualityScorer.SCORING_DIMENSIONS.items():
            score = EnhancedQualityScorer._score_dimension(document, config['checklist'])
            weighted_score = score * config['weight'] / 100
            total_score += weighted_score
            details[dimension] = {
                'score': score,
                'weighted': weighted_score
            }

        return {
            'total_score': round(total_score, 1),
            'grade': EnhancedQualityScorer._get_grade(total_score),
            'details': details
        }

    @staticmethod
    def _get_grade(score):
        if score >= 90:
            return 'A'
        elif score >= 80:
            return 'B'
        elif score >= 70:
            return 'C'
        else:
            return 'D'
```
