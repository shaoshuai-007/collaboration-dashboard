---

## 概述

本文档定义多智能体协同平台V15.2新增模块的完整API接口，涵盖产物生成、质量检查、模板管理三大核心模块。

### 基础信息

| 项目 | 值 |
|------|-----|
| Base URL | `https://api.multiagent.platform/v15.2` |
| 协议 | HTTPS |
| 数据格式 | JSON |
| 字符编码 | UTF-8 |
| 认证方式 | Bearer Token (JWT) |

---

### 通用错误码 (1xxx)

| 错误码 | HTTP状态码 | 说明 | 解决方案 |
|--------|-----------|------|---------|
| `INVALID_PARAMETER` | 400 | 参数验证失败 | 检查请求参数格式和取值 |

| `INTERNAL_ERROR` | 500 | 服务器内部错误 | 联系技术支持 |

```python
    def __init__(self, base_url: str, token: str):
        self.base_url = base_url.rstrip('/')
        self.session = requests.Session()
        self.session.headers.update({
            'Authorization': f'Bearer {token}',
            'Content-Type': 'application/json'
        })
    
    # ==================== 产物生成API ====================
    def generate_product(self, product_type: str, name: str, 
                        config: Optional[Dict] = None,
                        input_data: Optional[Dict] = None) -> Dict:
        """
        触发产物生成
        
        Args:
            product_type: 产物类型 (code/document/model/pipeline)
            name: 产物名称
            config: 生成配置
            input_data: 输入数据
            
        Returns:
            包含task_id的响应
        """
        url = f"{self.base_url}/api/product/generate"
        payload = {
            'type': product_type,
            'name': name,
            'config': config or {},
            'input_data': input_data or {}
        }
        
        response = self.session.post(url, json=payload)
        response.raise_for_status()
        return response.json()
    
    def get_product_status(self, task_id: str) -> Dict:
        """
        查询生成状态
        
        Args:
            task_id: 任务ID
            
        Returns:
            任务状态信息
        """
        url = f"{self.base_url}/api/product/status/{task_id}"
        response = self.session.get(url)
        response.raise_for_status()
        return response.json()
    
    def wait_for_completion(self, task_id: str, 
                           timeout: int = 3600,
                           poll_interval: int = 10) -> Dict:
        """
        等待任务完成
        
        Args:
            task_id: 任务ID
            timeout: 超时时间（秒）
            poll_interval: 轮询间隔（秒）
            
        Returns:
            最终状态信息
            
        Raises:
            TimeoutError: 超时
            RuntimeError: 任务失败
        """
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            status = self.get_product_status(task_id)
            
            if status['status'] == 'completed':
                return status
            elif status['status'] == 'failed':
                raise RuntimeError(f"任务失败: {status.get('error', {}).get('message')}")
            
            time.sleep(poll_interval)
        
        raise TimeoutError(f"任务超时: {task_id}")
    
    def download_product(self, task_id: str, 
                        output_path: str,
                        format: str = 'zip') -> str:
        """
        下载产物
        
        Args:
            task_id: 任务ID
            output_path: 输出文件路径
            format: 文件格式 (zip/tar.gz/raw)
            
        Returns:
            保存的文件路径
        """
        url = f"{self.base_url}/api/product/download/{task_id}"
        params = {'format': format}
        
        response = self.session.get(url, params=params, stream=True)
        response.raise_for_status()
        
        with open(output_path, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        
        return output_path
    
    # ==================== 质量检查API ====================
    def run_quality_check(self, target_id: str,
                         check_types: list,
                         target_type: str = 'product',
                         options: Optional[Dict] = None) -> Dict:
        """
        执行质量检查
        
        Args:
            target_id: 目标ID
            check_types: 检查类型列表
            target_type: 目标类型
            options: 检查选项
            
        Returns:
            检查任务信息
        """
        url = f"{self.base_url}/api/quality/check"
        payload = {
            'target_type': target_type,
            'target_id': target_id,
            'check_types': check_types,
            'options': options or {}
        }
        
        response = self.session.post(url, json=payload)
        response.raise_for_status()
        return response.json()
    
    def get_quality_report(self, check_id: str,
                          format: str = 'json') -> Dict:
        """
        获取质量报告
        
        Args:
            check_id: 检查任务ID
            format: 报告格式
            
        Returns:
            质量报告
        """
        url = f"{self.base_url}/api/quality/report/{check_id}"
        params = {'format': format}
        
        response = self.session.get(url, params=params)
        response.raise_for_status()
        
        if format == 'json':
            return response.json()
        return {'content': response.text}
    
    # ==================== 模板管理API ====================
    def list_templates(self, category: str = 'all',
                      search: Optional[str] = None,
                      tags: Optional[list] = None,
                      page: int = 1,
                      page_size: int = 20) -> Dict:
        """
        获取模板列表
        
        Args:
            category: 模板分类
            search: 搜索关键词
            tags: 标签过滤
            page: 页码
            page_size: 每页数量
            
        Returns:
            模板列表
        """
        url = f"{self.base_url}/api/templates"
        params = {
            'category': category,
            'page': page,
            'page_size': page_size
        }
        
        if search:
            params['search'] = search
        if tags:
            params['tags'] = ','.join(tags)
        
        response = self.session.get(url, params=params)
        response.raise_for_status()
        return response.json()

                       mode: str,
                       parameters: Optional[Dict] = None,
                       target_product_id: Optional[str] = None,
                       options: Optional[Dict] = None) -> Dict:

        应用模板
        
        Args:
            template_id: 模板ID
            mode: 应用模式 (create/merge/override)
            parameters: 模板参数
            target_product_id: 目标产物ID（merge/override模式）
            options: 应用选项
            
        Returns:
            任务信息
        """
        url = f"{self.base_url}/api/templates/apply"
        payload = {
            'template_id': template_id,
            'mode': mode,
            'parameters': parameters or {},
            'options': options or {}
        }
        
        if target_product_id:
            payload['target_product_id'] = target_product_id
        
        response = self.session.post(url, json=payload)
        response.raise_for_status()
        return response.json()
```

```typescript
class MultiAgentAPIClient {
  private baseUrl: string;
  private token: string;

  constructor(baseUrl: string, token: string) {
    this.baseUrl = baseUrl.replace(/\/$/, '');
    this.token = token;
  }

  private async request<T>(
    method: 'GET' | 'POST' | 'PUT' | 'DELETE',
    path: string,
    body?: any,
    params?: Record<string, string>
  ): Promise<T> {
    const url = new URL(`${this.baseUrl}${path}`);
    
    if (params) {
      Object.entries(params).forEach(([key, value]) => {
        url.searchParams.append(key, value);
      });
    }

    const response = await fetch(url.toString(), {
      method,
      headers: {
        'Authorization': `Bearer ${this.token}`,
        'Content-Type': 'application/json',
      },
      body: body ? JSON.stringify(body) : undefined,
    });

    if (!response.ok) {
      const error = await response.json();
      throw new APIError(error.code, error.message, error.details);
    }

    return response.json();
  }

  // ==================== 产物生成API ====================
  
  /**
   * 触发产物生成
   */
  async generateProduct(request: GenerateProductRequest): Promise<{
    task_id: string;
    status: string;
    message: string;
    status_url: string;
  }> {
    return this.request('POST', '/api/product/generate', request);
  }

  /**
   * 查询生成状态
   */
  async getProductStatus(taskId: string): Promise<ProductStatus> {
    return this.request('GET', `/api/product/status/${taskId}`);
  }

  /**
   * 等待任务完成
   */
  async waitForCompletion(
    taskId: string,
    timeout: number = 3600000, // 1小时
    pollInterval: number = 10000 // 10秒
  ): Promise<ProductStatus> {
    const startTime = Date.now();

    while (Date.now() - startTime < timeout) {
      const status = await this.getProductStatus(taskId);

      if (status.status === 'completed') {
        return status;
      }

      if (status.status === 'failed') {
        throw new Error(`任务失败: ${status.error?.message}`);
      }

      await new Promise(resolve => setTimeout(resolve, pollInterval));
    }

    throw new Error(`任务超时: ${taskId}`);
  }

  /**
   * 下载产物
   */
  async downloadProduct(
    taskId: string,
    format: 'zip' | 'tar.gz' | 'raw' = 'zip'
  ): Promise<Blob> {
    const url = new URL(`${this.baseUrl}/api/product/download/${taskId}`);
    url.searchParams.append('format', format);

    const response = await fetch(url.toString(), {
      headers: {
        'Authorization': `Bearer ${this.token}`,
      },
    });

    if (!response.ok) {
      const error = await response.json();
      throw new APIError(error.code, error.message, error.details);
    }

    return response.blob();
  }

  // ==================== 质量检查API ====================

  /**
   * 执行质量检查
   */
  async runQualityCheck(request: QualityCheckRequest): Promise<{
    check_id: string;
    status: string;
    message: string;
    report_url: string;
  }> {
    return this.request('POST', '/api/quality/check', request);
  }

  /**
   * 获取质量报告
   */
  async getQualityReport(
    checkId: string,
    format: 'json' | 'html' | 'pdf' | 'markdown' = 'json'
  ): Promise<QualityReport> {
    return this.request('GET', `/api/quality/report/${checkId}`, undefined, { format });
  }

  // ==================== 模板管理API ====================

  /**
   * 获取模板列表
   */
  async listTemplates(options: {
    category?: 'code' | 'document' | 'model' | 'pipeline' | 'all';
    search?: string;
    tags?: string[];
    page?: number;
    page_size?: number;
  } = {}): Promise<{
    templates: Template[];
    pagination: {
      page: number;
      page_size: number;
      total: number;
      total_pages: number;
    };
  }> {
    const params: Record<string, string> = {
      category: options.category || 'all',
      page: String(options.page || 1),
      page_size: String(options.page_size || 20),
    };

    if (options.search) params.search = options.search;
    if (options.tags) params.tags = options.tags.join(',');

    return this.request('GET', '/api/templates', undefined, params);
  }

  /**
   * 应用模板
   */
  async applyTemplate(request: ApplyTemplateRequest): Promise<{
    task_id: string;
    status: string;
    message: string;
    result_product_id?: string;
    status_url: string;
  }> {
    return this.request('POST', '/api/templates/apply', request);
  }
}

```

```
                ↓              ↓
```

### B. 速率限制策略

| API类型 | 限制 | 时间窗口 | 备注 |
|---------|------|---------|------|
| 标准API | 100次 | 1分钟 | 大部分读取接口 |
| 产物生成 | 10次 | 1分钟 | POST /api/product/generate |
| 文件下载 | 30次 | 1分钟 | GET /api/product/download/{id} |
| 质量检查 | 20次 | 1分钟 | POST /api/quality/check |
