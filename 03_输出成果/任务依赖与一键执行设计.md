```python
# 后端API

@app.route('/api/task/execute', methods=['POST'])
def execute_tasks():
    """执行任务链"""
    data = request.json
    tasks = data.get('tasks', [])  # 任务代码列表
    mode = data.get('mode', 'auto')  # auto/manual
    
    # 按依赖排序
    sorted_tasks = sort_by_dependencies(tasks)
    
    # 返回执行计划
    return jsonify({
        'plan': sorted_tasks,
        'total': len(sorted_tasks),
        'mode': mode
    })

@app.route('/api/task/next', methods=['POST'])
def next_task():
    """执行下一个任务"""
    data = request.json
    current_task = data.get('current_task')
    task_list = data.get('task_list')
    
    # 获取下一个任务
    next_index = task_list.index(current_task) + 1
    if next_index < len(task_list):
        next_task_code = task_list[next_index]
        # 触发Agent执行
        response = trigger_agent_task(next_task_code)
        return jsonify({
            'status': 'started',
            'task': next_task_code,
            'response': response
        })
    else:
        return jsonify({
            'status': 'completed',
            'message': '所有任务已完成'
        })
```

### 3.3 前端交互

```javascript
// 一键执行全部任务
async function executeAllTasks(tasks) {
    // 1. 获取执行计划
    const plan = await fetch('/api/task/execute', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({tasks: tasks, mode: 'auto'})
    }).then(r => r.json());
    
    // 2. 显示任务卡片
    showTaskCards(plan.plan);
    
    // 3. 逐个执行
    for (let i = 0; i < plan.plan.length; i++) {
        const task = plan.plan[i];
        updateTaskStatus(task.code, 'running');
        
        // 执行任务
        await executeTask(task.code);
        
        // 标记完成
        updateTaskStatus(task.code, 'completed');
        
        // 如果不是最后一个，显示"下一个任务"按钮
        if (i < plan.plan.length - 1) {
            showNextTaskButton(task.code);
        }
    }
}

// 执行单个任务
async function executeTask(taskCode) {
    const response = await fetch('/api/task/start', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({task_code: taskCode})
    });
    
    const result = await response.json();
    
    // 显示Agent讨论过程
    for (const message of result.discussion) {
        addMessage(message.agent, message.content);
    }
    
    // 显示产出物
    showOutput(result.output_template, result.output_content);
}
```

## 四、角色触发机制

### 4.1 根据RACI矩阵自动调度

```python
def trigger_agent_task(task_code: str) -> Dict:
    """
    根据RACI矩阵触发Agent执行任务
    
    流程:
    1. 获取任务的主导Agent（R角色）
    2. 获取参与Agent（A、C角色）
    3. 按顺序调用Agent
    4. 计算共识度
    5. 返回讨论结果和产出物
    """
    raci = RACI_MATRIX.get(task_code, {})
    
    # 主导Agent
    lead_agent = [a for a, r in raci.items() if r == 'R'][0]
    
    # 参与Agent
    participants = [a for a, r in raci.items() if r in ['R', 'A', 'C']]
    
    # 执行讨论
    discussion = []
    for agent_id in participants:
        response = generate_agent_response(agent_id, task_code)
        discussion.append({
            'agent': agent_id,
            'name': AGENTS[agent_id].name,
            'content': response
        })
    
    # 计算共识度
    consensus = calculate_consensus(discussion)
    
    # 生成产出物
    output = generate_output(task_code, discussion)
    
    return {
        'task_code': task_code,
        'discussion': discussion,
        'consensus': consensus,
        'output_template': TASK_TYPES[task_code].output_template,
        'output_content': output
    }
```
