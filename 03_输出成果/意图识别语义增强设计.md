```
用户输入: "需求分析"
```

```python
def _synonym_match(self, user_input: str) -> Tuple[str, float]:
    """第二层：同义词匹配"""
    for synonym, task_keywords in SYNONYM_DICT.items():
        if synonym in user_input:
            # 找到对应的任务
            for task_code, task in TASK_TYPES.items():
                for keyword in task_keywords:
                    if keyword in task.keywords:
                        return task_code, 0.85
    return None, 0.0
```

### 3.3 第三层：语义相似度计算

**基于 TF-IDF 的语义相似度**：

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class SemanticMatcher:
    def __init__(self):
        self.vectorizer = TfidfVectorizer()
        self.task_embeddings = None
        self._build_embeddings()
    
    def _build_embeddings(self):
        """构建任务描述的嵌入向量"""
        task_descriptions = []
        self.task_codes = []
        
        for task_code, task in TASK_TYPES.items():
            # 组合任务名称、描述、关键词
            text = f"{task.name} {task.description} {' '.join(task.keywords)}"
            task_descriptions.append(text)
            self.task_codes.append(task_code)
        
        # 训练 TF-IDF
        self.task_embeddings = self.vectorizer.fit_transform(task_descriptions)
    
    def calculate_similarity(self, user_input: str) -> List[Tuple[str, float]]:
        """计算用户输入与各任务的相似度"""
        # 向量化用户输入
        input_embedding = self.vectorizer.transform([user_input])
        
        # 计算余弦相似度
        similarities = cosine_similarity(input_embedding, self.task_embeddings)[0]
        
        # 返回按相似度排序的结果
        results = [(self.task_codes[i], similarities[i]) 
                   for i in range(len(self.task_codes))]
        results.sort(key=lambda x: x[1], reverse=True)
        
        return results

def _semantic_match(self, user_input: str) -> Tuple[str, float]:
    """第三层：语义相似度匹配"""
    matcher = SemanticMatcher()
    results = matcher.calculate_similarity(user_input)
    
    if results and results[0][1] >= 0.3:  # 相似度阈值
        return results[0][0], results[0][1] * 0.85  # 置信度归一化
    
    return None, 0.0
```

## 四、统一调度器

```python
class EnhancedIntentRecognizer:
    """增强版意图识别器"""
    
    def __init__(self):
        self.keyword_matcher = KeywordMatcher()
        self.synonym_matcher = SynonymMatcher()
        self.semantic_matcher = SemanticMatcher()
        self.llm_matcher = LLMMatcher()
    
    def recognize(self, user_input: str) -> Dict:
        """
        四层识别流程
        
        Returns:
            {
                'task_code': 任务代码,
                'task_name': 任务名称,
                'confidence': 置信度,
                'method': 识别方法,
                'alternatives': 备选任务
            }
        """
        # 第一层：精准关键词
        task_code, confidence = self.keyword_matcher.match(user_input)
        if confidence >= 0.90:
            return self._build_result(task_code, confidence, 'keyword_match')
        
        # 第二层：同义词
        task_code, confidence = self.synonym_matcher.match(user_input)
        if confidence >= 0.80:
            return self._build_result(task_code, confidence, 'synonym_match')
        
        # 第三层：语义相似度
        task_code, confidence, alternatives = self.semantic_matcher.match(user_input)
        if confidence >= 0.30:
            return self._build_result(
                task_code, confidence * 0.85, 'semantic_match', alternatives
            )
        
        # 第四层：大模型（可选，成本较高）
        if self._should_use_llm(user_input):
            task_code, confidence = self.llm_matcher.match(user_input)
            if confidence >= 0.85:
                return self._build_result(task_code, confidence, 'llm_match')
        
        # 默认：需求分析
        return self._build_result('REQ-02', 0.60, 'default')
    
    def _should_use_llm(self, user_input: str) -> bool:
        """判断是否应该调用大模型"""
        # 输入较短且前几层未命中，才调用
        return len(user_input) < 100
    
    def _build_result(self, task_code, confidence, method, alternatives=None):
        """构建返回结果"""
        task = TASK_TYPES.get(task_code)
        return {
            'task_code': task_code,
            'task_name': task.name if task else '未知',
            'confidence': confidence,
            'recognition_method': method,
            'alternatives': alternatives or []
        }
```

## 五、多任务识别增强

### 5.1 分句识别

```python
def recognize_multi(self, user_input: str) -> List[Dict]:
    """多任务识别"""
    
    # 1. 分句（按标点符号分割）
    sentences = re.split(r'[，,、；;和与及]', user_input)
    
    # 2. 对每个分句识别
    results = []
    seen_tasks = set()
    
    for sentence in sentences:
        sentence = sentence.strip()
        if not sentence:
            continue
        
        # 识别单个句子
        result = self.recognize(sentence)
        
        # 去重
        if result['task_code'] not in seen_tasks:
            seen_tasks.add(result['task_code'])
            results.append(result)
    
    return results
```

### 5.2 连词识别

```python
# 连词词典
CONNECTORS = {
    '和': 'AND',      # 并列关系
    '与': 'AND',
    '、': 'AND',
    '然后': 'THEN',   # 顺序关系
    '接着': 'THEN',
    '再': 'THEN',
    '同时': 'AND',    # 并行关系
    '并行': 'AND',
}

def parse_task_flow(self, user_input: str) -> List[Dict]:
    """解析任务流程（支持顺序/并行）"""
    
    tasks = []
    
    # 识别连词
    for connector, relation in CONNECTORS.items():
        if connector in user_input:
            parts = user_input.split(connector)
            for part in parts:
                result = self.recognize(part.strip())
                result['relation'] = relation
                tasks.append(result)
            return tasks
    
    # 无连词，直接识别
    return [self.recognize(user_input)]
```
