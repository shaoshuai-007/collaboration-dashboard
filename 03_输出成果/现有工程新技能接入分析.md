import subprocess

def process_intelligence(url):
    """处理情报URL"""
    # 1. 使用summarize摘要
    result = subprocess.run([
        'summarize', url, '--length', 'short'
    ], capture_output=True, text=True)
    
    # 2. 提取关键信息
    summary = result.stdout
    
    # 3. 推送给少帅
    return summary

def batch_process_intelligences(urls):
    """批量处理情报"""
    summaries = []
    for url in urls:
        summary = process_intelligence(url)
        summaries.append(summary)
    
    # 生成日报
    daily_brief = '\n\n---\n\n'.join(summaries)
    return daily_brief
```

# 文件：/root/.openclaw/workspace/02_工作台/文档服务/doc_summarizer.py

def summarize_document(doc_path):
    """摘要文档"""
    result = subprocess.run([
        'summarize', doc_path, '--length', 'medium'
    ], capture_output=True, text=True)
    
    return result.stdout

def summarize_meeting_recording(recording_url):
    """会议录音转纪要"""
    # 1. 转录
    transcript = subprocess.run([
        'summarize', recording_url, '--youtube', 'auto', '--extract-only'
    ], capture_output=True, text=True)
    
    # 2. 摘要
    minutes = subprocess.run([
        'summarize', '/dev/stdin', '--length', 'medium'
    ], input=transcript.stdout, capture_output=True, text=True)
    
    return minutes.stdout
```
