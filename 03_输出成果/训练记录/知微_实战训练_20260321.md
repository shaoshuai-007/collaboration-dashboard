"""
用户流失预警特征工程

import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# =====================================================
# 消费行为特征
# =====================================================

def calc_arpu_change_rate(df):
    """
    计算ARPU变化率
    
    Args:
        df: 用户数据DataFrame
    
    Returns:
        ARPU变化率Series
    """
    # 本月ARPU
    current_arpu = df['current_arpu']
    # 上月ARPU
    last_arpu = df['last_arpu']
    
    # 变化率 = (本月 - 上月) / 上月
    change_rate = (current_arpu - last_arpu) / last_arpu.replace(0, np.nan)
    
    return change_rate.fillna(0)

def calc_consumption_frequency_change(df):
    """
    计算消费频率变化率
    """
    current_freq = df['current_month_consumption_count']
    last_freq = df['last_month_consumption_count']
    
    change_rate = (current_freq - last_freq) / last_freq.replace(0, np.nan)
    return change_rate.fillna(0)

def calc_package_usage_rate(df):
    """
    计算套餐使用率
    """
    # 流量使用率
    data_usage = df['data_used'] / df['package_data_limit'].replace(0, np.nan)
    # 通话使用率
    call_usage = df['call_used'] / df['package_call_limit'].replace(0, np.nan)
    
    # 综合使用率（加权平均）
    usage_rate = (data_usage * 0.6 + call_usage * 0.4).fillna(0)
    
    # 截断到[0, 2]范围
    return usage_rate.clip(0, 2)

def calc_vas_penetration(df):
    """
    计算增值业务渗透率
    """
    return df['vas_count'] / df['total_service_count'].replace(0, np.nan).fillna(0)

def calc_recharge_interval(df):
    """
    计算充值间隔天数
    """
    # 最近一次充值距今天数
    return df['days_since_last_recharge']

def calc_arrears_count(df):
    """
    计算欠费次数（近3个月）
    """
    return df['arrears_count_3m']

# =====================================================
# 通信行为特征
# =====================================================

def calc_call_duration_change(df):
    """
    计算通话时长变化率
    """
    current_mou = df['current_mou']
    last_mou = df['last_mou']
    
    change_rate = (current_mou - last_mou) / last_mou.replace(0, np.nan)
    return change_rate.fillna(0)

def calc_data_usage_change(df):
    """
    计算流量使用变化率
    """
    current_dou = df['current_dou']
    last_dou = df['last_dou']
    
    change_rate = (current_dou - last_dou) / last_dou.replace(0, np.nan)
    return change_rate.fillna(0)

def calc_roaming_ratio(df):
    """
    计算漫游比例
    """
    return df['roaming_data'] / df['total_data'].replace(0, np.nan).fillna(0)

def calc_inter_network_ratio(df):
    """
    计算网间通话比例
    """
    return df['inter_network_call'] / df['total_call'].replace(0, np.nan).fillna(0)

def calc_service_consult_count(df):
    """
    计算客服咨询次数（近30天）
    """
    return df['service_call_count_30d']

def calc_complaint_count(df):
    """
    计算投诉次数（近30天）
    """
    return df['complaint_count_30d']

# =====================================================
# 交互行为特征
# =====================================================

def calc_app_active_days(df):
    """
    计算APP活跃天数（近30天）
    """
    return df['app_active_days_30d']

def calc_login_interval(df):
    """
    计算登录间隔天数
    """
    return df['days_since_last_login']

def calc_page_depth(df):
    """
    计算页面浏览深度
    """
    return df['avg_page_views_per_session']

def calc_feature_diversity(df):
    """
    计算功能使用多样性
    """
    # 使用的功能数 / 总功能数
    return df['used_features'] / df['total_features'].replace(0, np.nan).fillna(0)

def calc_marketing_engagement(df):
    """
    计算营销活动参与度
    """
    # 参与活动数 / 推送活动数
    return df['joined_activities'] / df['pushed_activities'].replace(0, np.nan).fillna(0)

def calc_points_exchange_freq(df):
    """
    计算积分兑换频率
    """
    return df['points_exchange_count_30d']

# =====================================================
# 特征汇总
# =====================================================

def build_churn_features(df):
    """
    构建流失预警特征
    
    Args:
        df: 原始用户数据DataFrame
    
    Returns:
        特征DataFrame
    """
    features = pd.DataFrame()
    
    # 用户ID
    features['user_id'] = df['user_id']
    
    # 消费行为特征
    features['arpu_change_rate'] = calc_arpu_change_rate(df)
    features['consumption_freq_change'] = calc_consumption_frequency_change(df)
    features['package_usage_rate'] = calc_package_usage_rate(df)
    features['vas_penetration'] = calc_vas_penetration(df)
    features['recharge_interval'] = calc_recharge_interval(df)
    features['arrears_count'] = calc_arrears_count(df)
    
    # 通信行为特征
    features['call_duration_change'] = calc_call_duration_change(df)
    features['data_usage_change'] = calc_data_usage_change(df)
    features['roaming_ratio'] = calc_roaming_ratio(df)
    features['inter_network_ratio'] = calc_inter_network_ratio(df)
    features['service_consult_count'] = calc_service_consult_count(df)
    features['complaint_count'] = calc_complaint_count(df)
    
    # 交互行为特征
    features['app_active_days'] = calc_app_active_days(df)
    features['login_interval'] = calc_login_interval(df)
    features['page_depth'] = calc_page_depth(df)
    features['feature_diversity'] = calc_feature_diversity(df)
    features['marketing_engagement'] = calc_marketing_engagement(df)
    features['points_exchange_freq'] = calc_points_exchange_freq(df)
    
    return features

def calc_feature_score(value, rules):
    """
    计算单个特征得分
    
    Args:
        value: 特征值
        rules: 评分规则
    
    Returns:
        得分和标签
    """
    for rule in rules:
        low, high = rule['range']
        if low <= value < high:
            return rule['score'], rule['label']
    
    return 0, '未知'

def calc_churn_score(features):
    """
    计算流失风险总分
    
    Args:
        features: 特征DataFrame
    
    Returns:
        评分结果DataFrame
    """
    results = []
    
    for idx, row in features.iterrows():
        total_score = 0
        score_details = []
        
        for feature_name, rules in CHURN_SCORE_RULES.items():
            if feature_name in row:
                value = row[feature_name]
                score, label = calc_feature_score(value, rules)
                total_score += score
                
                if score > 0:
                    score_details.append(f"{feature_name}: {label}({score}分)")
        
        # 风险等级判定
        if total_score >= 80:
            risk_level = '高风险'
            risk_color = '🔴'
        elif total_score >= 50:
            risk_level = '中风险'
            risk_color = '🟡'
        elif total_score >= 20:
            risk_level = '低风险'
            risk_color = '🟢'
        else:
            risk_level = '正常'
            risk_color = '✅'
        
        results.append({
            'user_id': row['user_id'],
            'churn_score': total_score,
            'risk_level': risk_level,
            'risk_color': risk_color,
            'score_details': '；'.join(score_details)
        })
    
    return pd.DataFrame(results)

        'priority': 'P2',

        'response_time': '一周内'

}

def generate_alert(churn_scores):
    """
    生成预警名单
    
    Args:
        churn_scores: 评分结果DataFrame
    
    Returns:
        预警名单DataFrame
    """
    alerts = []
    
    for level, rule in CHURN_ALERT_RULES.items():
        condition = rule['condition']
        filtered = churn_scores.query(condition)
        
        for _, row in filtered.iterrows():
            alerts.append({
                'alert_level': level,
                'priority': rule['priority'],
                'user_id': row['user_id'],
                'churn_score': row['churn_score'],
                'risk_details': row['score_details'],
                'action': rule['action'],
                'response_time': rule['response_time']
            })
    
    return pd.DataFrame(alerts).sort_values('churn_score', ascending=False)
