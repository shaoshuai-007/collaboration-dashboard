```python
"""
数据库模型定义
定义套餐表和套餐详情表的ORM模型
"""
from datetime import datetime
from sqlalchemy import Column, Integer, String, Text, DECIMAL, TIMESTAMP, ForeignKey
from sqlalchemy.orm import relationship
from database import Base


class Package(Base):
    """套餐表模型"""
    __tablename__ = "t_package"
    
    id = Column(Integer, primary_key=True, autoincrement=True, comment="套餐ID")
    name = Column(String(100), nullable=False, unique=True, comment="套餐名称")
    price = Column(DECIMAL(10, 2), nullable=False, comment="套餐价格")
    duration_months = Column(Integer, nullable=False, default=12, comment="有效期（月）")
    description = Column(Text, comment="套餐描述")
    category = Column(String(50), nullable=False, default="standard", comment="套餐类别")
    status = Column(Integer, nullable=False, default=1, comment="状态：1-启用，0-禁用")
    created_at = Column(TIMESTAMP, default=datetime.now, comment="创建时间")
    updated_at = Column(
        TIMESTAMP, 
        default=datetime.now, 
        onupdate=datetime.now, 
        comment="更新时间"
    )
    
    # 关系映射
    details = relationship(
        "PackageDetail", 
        back_populates="package",
        cascade="all, delete-orphan"
    )
    
    def __repr__(self):
        return f"<Package(id={self.id}, name='{self.name}', price={self.price})>"


class PackageDetail(Base):
    """套餐详情表模型"""
    __tablename__ = "t_package_detail"
    
    id = Column(Integer, primary_key=True, autoincrement=True, comment="详情ID")
    package_id = Column(Integer, ForeignKey("t_package.id"), nullable=False, comment="套餐ID")
    feature_name = Column(String(100), nullable=False, comment="功能名称")
    feature_value = Column(String(255), nullable=False, comment="功能值")
    unit = Column(String(20), comment="单位")
    description = Column(Text, comment="功能描述")
    created_at = Column(TIMESTAMP, default=datetime.now, comment="创建时间")
    
    # 关系映射
    package = relationship("Package", back_populates="details")
    
    def __repr__(self):
        return (
            f"<PackageDetail(id={self.id}, package_id={self.package_id}, "
            f"feature_name='{self.feature_name}')>"
        )
```

```python
"""
数据库CRUD操作
实现套餐的增删改查功能
"""
from decimal import Decimal
from typing import List, Optional
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from models import Package, PackageDetail
from schemas import (
    PackageCreate,
    PackageUpdate,
    PackageDetailCreate
)


class PackageCRUD:
    """套餐CRUD操作类"""
    
    @staticmethod
    def create_package(
        db: Session, 
        package: PackageCreate
    ) -> Package:
        """
        创建套餐
        
        Args:
            db: 数据库会话
            package: 套餐创建数据
            
        Returns:
            创建的套餐对象
        """
        # 创建套餐
        db_package = Package(
            name=package.name,
            price=package.price,
            duration_months=package.duration_months,
            description=package.description,
            category=package.category,
            status=package.status
        )
        db.add(db_package)
        db.flush()
        
        # 创建套餐详情
        if package.details:
            for detail in package.details:
                db_detail = PackageDetail(
                    package_id=db_package.id,
                    feature_name=detail.feature_name,
                    feature_value=detail.feature_value,
                    unit=detail.unit,
                    description=detail.description
                )
                db.add(db_detail)
        
        db.commit()
        db.refresh(db_package)
        return db_package
    
    @staticmethod
    def get_package(
        db: Session, 
        package_id: int
    ) -> Optional[Package]:
        """
        根据ID获取套餐
        
        Args:
            db: 数据库会话
            package_id: 套餐ID
            
        Returns:
            套餐对象或None
        """
        return db.query(Package).filter(
            Package.id == package_id
        ).first()
    
    @staticmethod
    def get_packages(
        db: Session,
        skip: int = 0,
        limit: int = 10,
        category: Optional[str] = None,
        status: Optional[int] = None,
        min_price: Optional[Decimal] = None,
        max_price: Optional[Decimal] = None
    ) -> tuple:
        """
        获取套餐列表（支持筛选和分页）
        
        Args:
            db: 数据库会话
            skip: 跳过记录数
            limit: 返回记录数
            category: 套餐类别筛选
            status: 状态筛选
            min_price: 最低价格筛选
            max_price: 最高价格筛选
            
        Returns:
            (总数, 套餐列表)
        """
        query = db.query(Package)
        
        # 应用筛选条件
        if category:
            query = query.filter(Package.category == category)
        if status is not None:
            query = query.filter(Package.status == status)
        if min_price is not None:
            query = query.filter(Package.price >= min_price)
        if max_price is not None:
            query = query.filter(Package.price <= max_price)
        
        # 获取总数
        total = query.count()
        
        # 分页查询
        packages = query.offset(skip).limit(limit).all()
        
        return total, packages
    
    @staticmethod
    def update_package(
        db: Session,
        package_id: int,
        package: PackageUpdate
    ) -> Optional[Package]:
        """
        更新套餐
        
        Args:
            db: 数据库会话
            package_id: 套餐ID
            package: 套餐更新数据
            
        Returns:
            更新后的套餐对象或None
        """
        db_package = db.query(Package).filter(
            Package.id == package_id
        ).first()
        
        if not db_package:
            return None
        
        # 更新字段
        update_data = package.model_dump(exclude_unset=True)
        for field, value in update_data.items():
            setattr(db_package, field, value)
        
        db.commit()
        db.refresh(db_package)
        return db_package
    
    @staticmethod
    def delete_package(
        db: Session, 
        package_id: int
    ) -> bool:
        """
        删除套餐
        
        Args:
            db: 数据库会话
            package_id: 套餐ID
            
        Returns:
            是否删除成功
        """
        db_package = db.query(Package).filter(
            Package.id == package_id
        ).first()
        
        if not db_package:
            return False
        
        db.delete(db_package)
        db.commit()
        return True
    
    @staticmethod
    def get_packages_by_ids(
        db: Session,
        package_ids: List[int]
    ) -> List[Package]:
        """
        根据ID列表获取多个套餐
        
        Args:
            db: 数据库会话
            package_ids: 套餐ID列表
            
        Returns:
            套餐列表
        """
        return db.query(Package).filter(
            Package.id.in_(package_ids)
        ).all()
    
    @staticmethod
    def recommend_packages(
        db: Session,
        budget: Optional[Decimal] = None,
        min_storage: Optional[int] = None,
        min_users: Optional[int] = None,
        features: Optional[List[str]] = None,
        limit: int = 3
    ) -> List[tuple]:
        """
        推荐套餐
        
        Args:
            db: 数据库会话
            budget: 预算上限
            min_storage: 最小存储空间
            min_users: 最小用户数
            features: 必需功能列表
            limit: 返回数量
            
        Returns:
            [(套餐对象, 匹配分数), ...]
        """
        # 查询所有启用的套餐
        packages = db.query(Package).filter(
            Package.status == 1
        ).all()
        
        results = []
        
        for package in packages:
            score = Decimal("100.00")
            match = True
            
            # 预算匹配
            if budget and package.price > budget:
                match = False
            elif budget:
                # 价格越接近预算，分数越高
                price_ratio = float(package.price / budget)
                score *= Decimal(str(min(price_ratio, 1.0)))
            
            # 存储空间匹配
            if min_storage:
                storage_detail = next(
                    (d for d in package.details if d.feature_name == "存储空间"),
                    None
                )
                if storage_detail:
                    try:
                        storage_value = int(storage_detail.feature_value)
                        if storage_value < min_storage:
                            match = False
                    except ValueError:
                        pass
            
            # 用户数匹配
            if min_users:
                users_detail = next(
                    (d for d in package.details if d.feature_name == "用户数"),
                    None
                )
                if users_detail:
                    try:
                        users_value = (
                            float('inf') 
                            if users_detail.feature_value == "不限" 
                            else int(users_detail.feature_value)
                        )
                        if users_value < min_users:
                            match = False
                    except ValueError:
                        pass
            
            # 功能匹配
            if features:
                package_features = {
                    d.feature_name for d in package.details
                }
                matched_features = set(features) & package_features
                feature_score = len(matched_features) / len(features)
                score *= Decimal(str(feature_score))
            
            if match:
                results.append((package, score))
        
        # 按匹配度排序
        results.sort(key=lambda x: x[1], reverse=True)
        
        return results[:limit]
```

```python
@router.post(
    "/compare",
    response_model=PackageCompareResponse,
    summary="套餐对比",
    description="对比多个套餐的功能特性"
)
def compare_packages(
    request: PackageCompareRequest,
    db: Session = Depends(get_db)
):
    """
    套餐对比
    
    对比2-5个套餐的功能特性
    
    - **package_ids**: 要对比的套餐ID列表（2-5个）
    """
    # 获取套餐列表
    packages = PackageCRUD.get_packages_by_ids(db, request.package_ids)
    
    if len(packages) != len(request.package_ids):
        raise HTTPException(
            status_code=400,
            detail="部分套餐不存在"
        )
    
    # 收集所有功能名称
    all_features = set()
    for package in packages:
        for detail in package.details:
            all_features.add(detail.feature_name)
    
    # 构建对比结果
    features_comparison = []
    for feature_name in sorted(all_features):
        values = {}
        for package in packages:
            detail = next(
                (d for d in package.details if d.feature_name == feature_name),
                None
            )
            if detail:
                value = detail.feature_value
                if detail.unit:
                    value = f"{value}{detail.unit}"
                values[package.name] = value
            else:
                values[package.name] = "不支持"
        
        features_comparison.append(
            FeatureComparison(feature_name=feature_name, values=values)
        )
    
    return PackageCompareResponse(
        packages=packages,
        features=features_comparison
    )

    - **budget**: 预算上限
    - **min_storage**: 最小存储空间（GB）
    - **min_users**: 最小用户数
    - **features**: 必需功能列表
    """

        db,
        budget=request.budget,
        min_storage=request.min_storage,
        min_users=request.min_users,
        features=request.features,
        limit=1
    )
    
    if not results:
        raise HTTPException(
            status_code=404,
            detail="未找到符合条件的套餐"
        )
    
    recommended_package, score = results[0]
    
    # 生成推荐理由
    reasons = []
    if request.budget:
        reasons.append(f"价格在预算范围内（¥{recommended_package.price}）")
    if request.min_storage:
        reasons.append(f"满足存储需求")
    if request.min_users:
        reasons.append(f"满足用户数需求")
    if request.features:
        reasons.append(f"包含所需功能")
    
    if not reasons:
        reasons.append("综合性价比最优")
    
    return RecommendResponse(
        recommended=recommended_package,
        score=score,
        reasons=reasons
    )
```

```python
class PackageCRUD:
    """套餐CRUD操作类"""
    
    @staticmethod
    def create_package(db: Session, package: PackageCreate) -> Package:
        """创建套餐"""
        # 创建套餐
        db_package = Package(
            name=package.name,
            price=package.price
        )
        db.add(db_package)
        db.flush()
        
        # 创建套餐详情
        if package.details:
            for detail in package.details:
                db_detail = PackageDetail(
                    package_id=db_package.id,
                    feature_name=detail.feature_name
                )
                db.add(db_detail)
        
        db.commit()
        db.refresh(db_package)
        
        return db_package
    
    @staticmethod
    def get_package(db: Session, package_id: int) -> Optional[Package]:
        """获取套餐"""
        return db.query(Package).filter(
            Package.id == package_id
        ).first()
```
