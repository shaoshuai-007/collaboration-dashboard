### 2.1 分析方法复习（30分钟）

#### RFM模型

**定义**：RFM模型是衡量客户价值和客户创利能力的重要工具和手段。

| 维度 | 含义 | 指标 |
|:----:|------|------|
| R | Recency（最近一次消费） | 最近一次消费距今天数 |
| F | Frequency（消费频率） | 消费次数 |
| M | Monetary（消费金额） | 消费总额 |

**用户分群**：

| 用户类型 | R | F | M | 特征 |
|----------|:-:|:-:|:-:|------|
| 高价值用户 | ↑ | ↑ | ↑ | 最近消费、频率高、金额大 |
| 潜力用户 | ↓ | ↑ | ↑ | 消费频繁、金额大但已有一段时间未消费 |
| 新用户 | ↑ | ↓ | ↓ | 最近消费但频率低、金额小 |
| 流失预警用户 | ↓ | ↓ | ↓ | 很久未消费、频率低、金额小 |
| 低价值用户 | ↓ | ↓ | ↓ | 消费间隔长、频率低、金额小 |

#### 用户画像构建方法

**画像维度**：

```
基础属性
├── 人口统计：年龄、性别、地域
└── 消费能力：ARPU、消费频率

行为特征
├── 通信行为：通话、流量、短信
└── 业务偏好：套餐类型、增值业务

兴趣标签
├── 应用偏好：视频、游戏、社交
└── 内容偏好：娱乐、新闻、教育

生命周期
├── 新用户期
├── 成长期
├── 成熟期
├── 衰退期
└── 流失期
```

**标签规则示例**：

| 标签 | 规则 | 计算 |
|------|------|------|
| 高价值用户 | ARPU > 200 且 活跃天数 > 25 | `arpu > 200 AND active_days > 25` |
| 流量大户 | DOU > 10GB | `doug > 10240` |
| 通话达人 | MOU > 500分钟 | `mou > 500` |
| 流失预警 | 最近30天无消费 | `last_consumption_days > 30` |

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from datetime import datetime, timedelta

# 设置中文显示
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# =====================================================
# 数据加载与清洗
# =====================================================

def load_and_clean_data(file_path):
    """
    加载并清洗数据

    Args:
        file_path: 数据文件路径

    Returns:
        清洗后的DataFrame
    """
    # 加载数据
    df = pd.read_csv(file_path)

    # 检查数据基本信息
    print(f"数据量: {len(df)} 条")
    print(f"字段数: {len(df.columns)} 个")

    # 缺失值处理
    missing = df.isnull().sum()
    print(f"缺失值统计:\n{missing[missing > 0]}")

    # 填充缺失值
    df['arpu'].fillna(df['arpu'].median(), inplace=True)
    df['dou'].fillna(0, inplace=True)
    df['mou'].fillna(0, inplace=True)

    # 异常值处理
    df = df[df['arpu'] > 0]  # ARPU必须大于0
    df = df[df['arpu'] < 1000]  # 过滤极端异常值

    return df

# =====================================================
# ARPU分析
# =====================================================

def analyze_arpu_distribution(df):
    """
    分析ARPU分布

    Args:
        df: 用户数据DataFrame

    Returns:
        ARPU分布统计结果
    """
    # ARPU区间划分
    bins = [0, 50, 100, 200, 300, float('inf')]
    labels = ['0-50', '50-100', '100-200', '200-300', '300+']

    df['arpu_range'] = pd.cut(df['arpu'], bins=bins, labels=labels)

    # 统计各区间用户数
    arpu_dist = df['arpu_range'].value_counts().sort_index()

    # 计算占比
    arpu_pct = (arpu_dist / len(df) * 100).round(2)

    # 汇总结果
    result = pd.DataFrame({
        '用户数': arpu_dist,
        '占比(%)': arpu_pct
    })

    return result

def plot_arpu_distribution(arpu_dist):
    """
    可视化ARPU分布

    Args:
        arpu_dist: ARPU分布统计结果
    """
    fig, axes = plt.subplots(1, 2, figsize=(14, 6))

    # 柱状图
    colors = ['#006EBD', '#006EBD', '#C93832', '#C93832', '#595959']
    axes[0].bar(arpu_dist.index, arpu_dist['用户数'], color=colors)
    axes[0].set_title('ARPU区间用户分布', fontsize=14, fontweight='bold')
    axes[0].set_xlabel('ARPU区间（元）', fontsize=12)
    axes[0].set_ylabel('用户数', fontsize=12)

    # 添加数据标签
    for i, v in enumerate(arpu_dist['用户数']):
        axes[0].text(i, v + 1000, f'{v:,}', ha='center', va='bottom')

    # 饼图
    axes[1].pie(arpu_dist['占比(%)'], labels=arpu_dist.index, autopct='%1.1f%%',
                colors=colors, startangle=90)
    axes[1].set_title('ARPU区间占比', fontsize=14, fontweight='bold')

    plt.tight_layout()
    plt.savefig('arpu_distribution.png', dpi=150, bbox_inches='tight')
    plt.close()

    print("图表已保存: arpu_distribution.png")

# =====================================================
# RFM分群分析
# =====================================================

def rfm_analysis(df):
    """
    RFM分群分析

    Args:
        df: 用户数据DataFrame

    Returns:
        RFM分群结果
    """
    # 计算RFM值
    reference_date = datetime.now()

    df['R'] = (reference_date - pd.to_datetime(df['last_consumption_date'])).dt.days
    df['F'] = df['consumption_count']
    df['M'] = df['total_consumption']

    # RFM评分（5分制）
    df['R_score'] = pd.qcut(df['R'], 5, labels=[5, 4, 3, 2, 1], duplicates='drop')
    df['F_score'] = pd.qcut(df['F'], 5, labels=[1, 2, 3, 4, 5], duplicates='drop')
    df['M_score'] = pd.qcut(df['M'], 5, labels=[1, 2, 3, 4, 5], duplicates='drop')

    # 计算RFM总分
    df['RFM_score'] = df['R_score'].astype(int) + df['F_score'].astype(int) + df['M_score'].astype(int)

    # 用户分群
    def classify_user(row):
        r, f, m = row['R_score'], row['F_score'], row['M_score']

        if r >= 4 and f >= 4 and m >= 4:
            return '高价值用户'
        elif r <= 2 and f >= 4 and m >= 4:
            return '潜力用户'
        elif r >= 4 and f <= 2 and m <= 2:
            return '新用户'
        elif r <= 2 and f <= 2 and m <= 2:
            return '流失预警用户'
        else:
            return '普通用户'

    df['user_segment'] = df.apply(classify_user, axis=1)

    # 统计各群体
    segment_stats = df.groupby('user_segment').agg({
        'user_id': 'count',
        'arpu': 'mean',
        'R': 'mean',
        'F': 'mean',
        'M': 'mean'
    }).round(2)

    segment_stats.columns = ['用户数', '平均ARPU', '平均R', '平均F', '平均M']

    return df, segment_stats

def build_user_profile(df, user_id):
    """
    构建单个用户画像

    Args:
        df: 用户数据DataFrame
        user_id: 用户ID

    Returns:
        用户画像字典
    """
    user = df[df['user_id'] == user_id].iloc[0]

    profile = {
        # 基础属性
        'user_id': user_id,
        'phone': user['phone'],
        'region': user['region'],
        'age_group': user['age_group'],

        # 消费特征
        'arpu': user['arpu'],
        'arpu_level': '高' if user['arpu'] > 200 else ('中' if user['arpu'] > 100 else '低'),
        'consumption_trend': '上升' if user['consumption_change'] > 0.1 else ('下降' if user['consumption_change'] < -0.1 else '稳定'),

        # 行为特征
        'dou': user['dou'],
        'mou': user['mou'],
        'dou_level': '高' if user['dou'] > 10240 else ('中' if user['dou'] > 5120 else '低'),
        'mou_level': '高' if user['mou'] > 500 else ('中' if user['mou'] > 200 else '低'),

        # 价值分群
        'rfm_segment': user['user_segment'],
        'rfm_score': user['RFM_score'],

        # 标签
        'tags': []
    }

    # 生成标签
    if profile['arpu_level'] == '高':
        profile['tags'].append('高价值用户')
    if profile['dou_level'] == '高':
        profile['tags'].append('流量大户')
    if profile['mou_level'] == '高':
        profile['tags'].append('通话达人')
    if user['last_consumption_days'] > 30:
        profile['tags'].append('流失预警')

    return profile

def generate_segment_profiles(df):
    """
    生成分群画像

    Args:
        df: 用户数据DataFrame

    Returns:
        分群画像报告
    """
    segments = df['user_segment'].unique()
    profiles = {}

    for segment in segments:
        segment_df = df[df['user_segment'] == segment]

        profiles[segment] = {
            '用户数量': len(segment_df),
            '占比': f"{len(segment_df) / len(df) * 100:.1f}%",
            '平均ARPU': f"{segment_df['arpu'].mean():.0f}元",
            '平均DOU': f"{segment_df['dou'].mean() / 1024:.1f}GB",
            '平均MOU': f"{segment_df['mou'].mean():.0f}分钟",
            '特征描述': get_segment_description(segment)
        }

    return profiles

def get_segment_description(segment):
    """获取分群描述"""
    descriptions = {
        '高价值用户': '最近消费活跃、消费频率高、消费金额大，是核心优质客户',
        '潜力用户': '消费频率和金额较高但有一段时间未消费，需要激活',
        '新用户': '最近有消费但频率和金额较低，需要培育',
        '流失预警用户': '消费间隔长、频率低、金额低，有流失风险',
        '普通用户': '各项指标中等，有一定消费潜力'
    }
    return descriptions.get(segment, '')
