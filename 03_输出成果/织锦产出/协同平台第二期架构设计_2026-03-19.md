### 3.2 核心数据结构

```python
# 1. 结构化意图
class StructuredIntent:
    project_name: str           # 项目名称
    task_type: str              # 任务类型（REQ-02等）
    outputs: List[Output]       # 产出物列表
    complexity: str             # 复杂度（low/medium/high）
    participants: List[str]     # 参与者
    discussion_rounds: int      # 讨论轮次
    deadline: datetime          # 截止时间

# 2. 讨论结果
class DiscussionResult:
    rounds: List[Round]         # 各轮讨论
    consensus_reached: bool     # 是否达成共识
    consensus: str              # 共识内容
    decisions: List[Decision]   # 关键决策
    meeting_minutes: str        # 会议纪要（Markdown）

# 3. 产物蓝图
class OutputBlueprint:
    output_type: str            # 产物类型（SRS/SAD/DDD等）
    output_name: str            # 产物名称
    sections: List[Section]     # 章节规划
    dependencies: List[str]     # 依赖的前序产物
    template: str               # 使用的模板

# 4. 任务清单
class TaskAssignment:
    agent: str                  # 执行Agent
    task: str                   # 任务描述
    sections: List[str]         # 负责章节
    input_context: str          # 输入上下文
    expected_output: str        # 期望输出
    status: str                 # 状态（pending/running/done）

# 5. 产物版本
class OutputVersion:
    version: str                # 版本号（v1.0.0）
    content: str                # 内容（Markdown）
    changelog: List[str]        # 变更日志
    created_at: datetime        # 创建时间
    created_by: List[str]       # 创建者（Agent列表）
    quality_score: float        # 质量分数
```

### 4.2 模块1：产物生成引擎

#### 4.2.1 设计目标

将讨论结果转化为具体的专业产物（文档/代码/方案）。

#### 4.2.2 核心组件

```python
class OutputGenerationEngine:
    """产物生成引擎"""
    
    def __init__(self):
        self.blueprint_generator = BlueprintGenerator()
        self.task_dispatcher = TaskDispatcher()
        self.result_integrator = ResultIntegrator()
    
    def generate(self, discussion_result: DiscussionResult, 
                 intent: StructuredIntent) -> OutputResult:
        """
        从讨论结果生成产物
        
        流程:
        1. 产物蓝图生成
        2. 任务分解与分配
        3. 并行执行
        4. 结果整合
        5. 质量检查
        """
        # 1. 生成产物蓝图
        blueprint = self.blueprint_generator.generate(
            discussion_result, intent
        )
        
        # 2. 任务分解
        tasks = self.task_dispatcher.decompose(blueprint)
        
        # 3. 任务分配（基于岗位专属原则）
        assignments = self.task_dispatcher.assign(tasks, intent.participants)
        
        # 4. 并行执行
        results = await self.task_dispatcher.execute_parallel(assignments)
        
        # 5. 结果整合
        output = self.result_integrator.integrate(results, blueprint)
        
        return output
```

```python
                sections=sections,

                template=template['name']
            )
            blueprints.append(blueprint)
        
        return blueprints
```

#### 4.2.4 任务调度器

```python
class TaskDispatcher:
    """任务调度器"""
    
    def decompose(self, blueprint: OutputBlueprint) -> List[Task]:
        """任务分解"""
        tasks = []
        for section in blueprint.sections:
            task = Task(
                section_name=section['name'],
                owner=section['owner'],
                description=self._generate_description(section),
                input_context=self._get_context(section),
                status='pending'
            )
            tasks.append(task)
        return tasks
    
    def assign(self, tasks: List[Task], participants: List[str]) -> List[TaskAssignment]:
        """
        任务分配（遵循岗位专属原则）
        
        规则：
        - 采薇 → 需求相关章节
        - 织锦 → 架构相关章节
        - 工尺 → 系统/接口设计章节
        - 天工 → 代码实现章节
        - 玉衡 → 项目管理章节
        - 筑台 → 成本/售前章节
        """
        assignments = []
        for task in tasks:
            # 确定执行者（基于岗位专属原则）
            owner = self._determine_owner(task, participants)
            
            assignment = TaskAssignment(
                agent=owner,
                task=task.description,
                sections=[task.section_name],
                input_context=task.input_context,
                expected_output=self._get_expected_output(task),
                status='pending'
            )
            assignments.append(assignment)
        
        return assignments
    
    async def execute_parallel(self, assignments: List[TaskAssignment]) -> List[Output]:
        """并行执行任务"""
        # 使用asyncio并行执行
        tasks = [
            self._execute_single(assignment) 
            for assignment in assignments
        ]
        results = await asyncio.gather(*tasks)
        return results
    
    async def _execute_single(self, assignment: TaskAssignment) -> Output:
        """执行单个任务"""
        # 调用对应Agent的专属技能
        agent = self._get_agent(assignment.agent)
        skill = self._get_skill(assignment.agent)
        
        output = await agent.execute_with_skill(
            skill=skill,
            task=assignment.task,
            context=assignment.input_context
        )
        
        assignment.status = 'done'
        return output
```

#### 4.2.5 结果整合器

```python
class ResultIntegrator:
    """结果整合器"""
    
    def integrate(self, results: List[Output], 
                  blueprint: OutputBlueprint) -> OutputResult:
        """整合各Agent产出"""
        
        # 1. 按章节顺序组装
        sections_content = {}
        for result in results:
            sections_content[result.section_name] = result.content
        
        # 2. 生成完整文档
        full_content = self._assemble_document(
            sections_content, blueprint
        )
        
        # 3. 格式化处理
        formatted_content = self._format_document(full_content)
        
        return OutputResult(
            output_type=blueprint.output_type,
            content=formatted_content,
            sections=sections_content,
            created_by=list(set(r.agent for r in results))
        )
    
    def _assemble_document(self, sections: dict, 
                           blueprint: OutputBlueprint) -> str:
        """组装文档"""
        document = f"# {blueprint.output_name}\n\n"
        document += f"> 生成时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}\n"
        document += f"> 参与者: {', '.join(blueprint.participants)}\n\n"
        
        for section in blueprint.sections:
            document += f"## {section['name']}\n\n"
            document += sections.get(section['name'], '[待补充]')
            document += "\n\n"
        
        return document
```

```python
    """专家评审器"""
    
    # 评审专家映射（基于岗位专属原则）
    REVIEWER_MAP = {
        'SRS': ['采薇', '玉衡'],      # 需求规格：采薇主导，玉衡确认
        'SAD': ['织锦', '工尺'],      # 架构设计：织锦主导，工尺确认
        'DDD': ['工尺', '天工'],      # 详细设计：工尺主导，天工确认
        'PP': ['玉衡', '筑台'],       # 项目计划：玉衡主导，筑台确认
    }
    
    def review(self, output: OutputResult) -> ReviewResult:
        """专家评审"""
        reviewers = self.REVIEWER_MAP.get(output.output_type, [])
        
        reviews = []
        for reviewer in reviewers:
            review = self._conduct_review(reviewer, output)
            reviews.append(review)
        
        return ReviewResult(
            output_id=output.id,
            reviews=reviews,
            consensus=self._get_consensus(reviews)
        )
    
    def _conduct_review(self, reviewer: str, 
                        output: OutputResult) -> Review:
        """执行评审"""
        agent = self._get_agent(reviewer)
        
        # 调用Agent的评审能力
        review_comments = agent.review_output(
            output=output,
            focus_areas=self._get_focus_areas(reviewer)
        )
        
        return Review(
            reviewer=reviewer,
            comments=review_comments,
            rating=self._calculate_rating(review_comments)
        )
```

```python
class VersionManagementEngine:
    """版本管理引擎"""
    
    def __init__(self):
        self.version_store = VersionStore()
        self.change_tracker = ChangeTracker()
        self.rollback_manager = RollbackManager()
    
    def save_version(self, output: OutputResult, 
                     message: str = "") -> Version:
        """保存版本"""
        # 生成版本号
        version_number = self._generate_version_number(output.id)
        
        # 记录变更
        changes = self.change_tracker.diff(
            self.version_store.get_latest(output.id),
            output
        )
        
        # 保存版本
        version = Version(
            id=str(uuid.uuid4()),
            output_id=output.id,
            version=version_number,
            content=output.content,
            changes=changes,
            message=message,
            created_at=datetime.now(),
            created_by=output.created_by
        )
        
        self.version_store.save(version)
        return version
    
    def get_history(self, output_id: str) -> List[Version]:
        """获取版本历史"""
        return self.version_store.get_history(output_id)
    
    def rollback(self, output_id: str, 
                 target_version: str) -> OutputResult:
        """版本回滚"""
        return self.rollback_manager.rollback(output_id, target_version)
    
    def diff_versions(self, output_id: str, 
                      v1: str, v2: str) -> DiffResult:
        """版本对比"""
        version1 = self.version_store.get(output_id, v1)
        version2 = self.version_store.get(output_id, v2)
        return self.change_tracker.diff(version1, version2)
```

### 5.1 技术选型总览

| 层次 | 技术选型 | 选型理由 |
|------|---------|---------|
| **协调层** | Python + FastAPI | 与现有V15平台一致，降低迁移成本 |
| **异步执行** | asyncio + Celery | 支持并行执行和任务队列 |

| **缓存** | Redis | 提升访问性能 |
| **监控** | Prometheus + Grafana | 完善的监控告警体系 |

### 5.2 核心技术选型详解

#### 5.2.1 异步执行框架

**选择**: asyncio + Celery

**理由**:
1. asyncio支持协程，减少线程开销
2. Celery支持分布式任务队列
3. 与现有Python技术栈兼容

**架构**:
```
任务提交 → Celery队列 → Worker执行 → 结果回调
         ↓
    Redis (消息代理)
```

#### 5.2.2 知识存储方案

**选择**: Milvus向量数据库

**理由**:
1. 支持高维向量检索
2. 开源，部署简单
3. 支持混合检索（向量+标量）

**架构**:
```
知识入库 → Embedding编码 → 向量存储 → 语义检索
                              ↓
                          Milvus集群
```

#### 5.2.3 版本管理方案

**选择**: Git + 本地文件系统

**理由**:
1. Git天然支持版本对比和回滚
2. 成熟稳定，生态完善
3. 支持分支管理

**架构**:
```
产物保存 → Git Commit → 版本快照 → 变更日志
            ↓
        本地仓库
```
