# 质疑增强机制方案 V1.0

**生成时间**：2026-03-15 20:05  
**目标**：提升质疑机制的主动性、深度、可追溯性

---

## 一、现状分析

### 当前质疑机制

| 组件 | 实现方式 | 评估 |
|------|---------|------|
| 触发机制 | 关键词检测（质疑、反对、不同意等） | 被动，可能遗漏 |
| 质疑标记 | `is_challenging=True`，显示徽章 | 基础，缺少分类 |
| 共识度 | 基础70%，每条质疑扣5分 | 简单，不精确 |
| 辩论环节 | 固定阶段 | 可优化 |

### 存在的问题

1. **质疑触发不够主动**：依赖Agent主动提出，可能遗漏关键风险
2. **质疑深度不够**：简单的关键词检测，可能误判或漏判
3. **质疑回应缺失**：被质疑的Agent没有强制回应机制
4. **质疑可视化不足**：用户看不到质疑链路（谁质疑谁，如何解决）
5. **质疑强度评估缺失**：没有区分轻度建议和重度反对

```python
class ConflictDetector:
    """观点冲突检测器"""
    
    CONFLICT_PATTERNS = [
        # 技术路线冲突
        (r'微服务', r'单体架构'),
        (r'自研', r'外包'),
        (r'开源', r'商业'),
        
        # 成本冲突
        (r'成本可控', r'成本过高'),
        (r'预算充足', r'预算不足'),
        
        # 工期冲突
        (r'工期充足', r'工期紧张'),
        (r'可以完成', r'无法完成'),
    ]
    
    def detect_conflict(self, history):
        """检测观点冲突"""
        conflicts = []
        for pattern_a, pattern_b in self.CONFLICT_PATTERNS:
            # 找出支持A和反对B的发言
            supporters_a = []
            supporters_b = []
            
            for msg in history:
                if re.search(pattern_a, msg.content):
                    supporters_a.append(msg.speaker)
                if re.search(pattern_b, msg.content):
                    supporters_b.append(msg.speaker)
            
            if supporters_a and supporters_b:
                conflicts.append({
                    'type': '观点冲突',
                    'pattern': f'{pattern_a} vs {pattern_b}',
                    'sides': {
                        'A': supporters_a,
                        'B': supporters_b
                    }
                })
        
        return conflicts
```

@dataclass
class StructuredChallenge:
    """结构化质疑"""
    
    challenge_id: str                   # 质疑ID
    challenger: str                     # 质疑者
    target_agent: str                   # 被质疑者
    target_point: str                   # 被质疑的观点
    
    category: ChallengeCategory         # 质疑类型
    intensity: ChallengeIntensity       # 质疑强度
    
    reason: str                         # 质疑原因
    evidence: List[str]                 # 支持证据
    suggestion: str                     # 改进建议
    
    status: str = "pending"             # pending/accepted/rejected/resolved
    response: str = ""                  # 回应内容
    
    def to_prompt(self):
        """生成质疑提示"""
        return f"""
🎯 质疑 #{self.challenge_id}

**质疑者**：{self.challenger}
**被质疑者**：{self.target_agent}
**质疑观点**：{self.target_point}

**质疑类型**：{self.category.value}
**质疑强度**：{'⚠️' * self.intensity.value} {self.intensity.name}

**质疑原因**：
{self.reason}

**支持证据**：
{chr(10).join(f'- {e}' for e in self.evidence)}

**改进建议**：
{self.suggestion}
"""

### 2.3 质疑回应机制

#### A. 强制回应

```python
class ChallengeResponseHandler:
    """质疑回应处理器"""
    
    def __init__(self, memory):
        self.memory = memory
        self.pending_challenges = []
    
    def register_challenge(self, challenge: StructuredChallenge):
        """注册质疑"""
        self.pending_challenges.append(challenge)
        
        # 通知被质疑者
        target_agent = challenge.target_agent
        prompt = f"""
📢 你被质疑了！

{challenge.to_prompt()}

请从你的专业角度回应这个质疑：
1. 接受质疑并调整方案
2. 解释原因，证明原方案合理
3. 提出折中方案

**注意**：必须在下一轮发言中回应此质疑！
"""
        return prompt
    
    def check_response(self, agent_id, response_content):
        """检查是否回应了质疑"""
        for challenge in self.pending_challenges:
            if challenge.target_agent == agent_id and challenge.status == "pending":
                # 检测回应类型
                if any(kw in response_content for kw in ['接受', '同意', '认可']):
                    challenge.status = "accepted"
                    challenge.response = response_content
                elif any(kw in response_content for kw in ['不同意', '反驳', '坚持']):
                    challenge.status = "rejected"
                    challenge.response = response_content
                else:
                    challenge.status = "resolved"
                    challenge.response = response_content
                
                return True
        return False
```

#### B. 质疑链路追踪

```python
class ChallengeChain:
    """质疑链路"""
    
    def __init__(self):
        self.chains = []  # [{challenge, responses, resolution}]
    
    def add_challenge(self, challenge: StructuredChallenge):
        """添加质疑"""
        self.chains.append({
            'challenge': challenge,
            'responses': [],
            'resolution': None
        })
    
    def add_response(self, challenge_id, agent_id, response, accepts):
        """添加回应"""
        for chain in self.chains:
            if chain['challenge'].challenge_id == challenge_id:
                chain['responses'].append({
                    'agent': agent_id,
                    'content': response,
                    'accepts': accepts,
                    'timestamp': datetime.now()
                })
                break
    
    def resolve(self, challenge_id, resolution):
        """解决质疑"""
        for chain in self.chains:
            if chain['challenge'].challenge_id == challenge_id:
                chain['resolution'] = resolution
                chain['challenge'].status = "resolved"
                break
    
    def to_summary(self):
        """生成质疑总结"""
        summary = "## 质疑链路总结\n\n"
        
        for i, chain in enumerate(self.chains, 1):
            challenge = chain['challenge']
            summary += f"### 质疑 {i}\n\n"
            summary += f"**质疑者**：{challenge.challenger}\n"
            summary += f"**被质疑者**：{challenge.target_agent}\n"
            summary += f"**质疑类型**：{challenge.category.value}\n"
            summary += f"**质疑强度**：{'⚠️' * challenge.intensity.value}\n\n"
            summary += f"**质疑内容**：{challenge.reason}\n\n"
            
            if chain['responses']:
                summary += "**回应记录**：\n"
                for resp in chain['responses']:
                    status = "✅ 接受" if resp['accepts'] else "❌ 拒绝"
                    summary += f"- {resp['agent']}: {status}\n"
            
            if chain['resolution']:
                summary += f"\n**解决结果**：{chain['resolution']}\n"
            
            summary += "\n---\n\n"
        
        return summary
```

### 2.5 共识度精细化

#### A. 多维度共识度

```python
class ConsensusCalculator:
    """共识度计算器"""
    
    def calculate_detailed_consensus(self, history):
        """计算多维度共识度"""
        
        # 各维度共识度
        dimensions = {
            'requirement': 100,    # 需求共识
            'technical': 100,     # 技术共识
            'cost': 100,          # 成本共识
            'schedule': 100,      # 工期共识
            'risk': 100,          # 风险共识
        }
        
        # 按维度扣分
        for msg in history:
            if msg.is_challenging:
                challenge = msg.challenge_details
                dimension = self.map_category_to_dimension(challenge.category)
                penalty = INTENSITY_PENALTY[challenge.intensity]
                dimensions[dimension] = max(0, dimensions[dimension] - penalty)
        
        # 总体共识度（加权平均）
        weights = {
            'requirement': 0.2,
            'technical': 0.25,
            'cost': 0.2,
            'schedule': 0.2,
            'risk': 0.15,
        }
        
        total_consensus = sum(
            dimensions[k] * weights[k] for k in dimensions
        )
        
        return {
            'total': total_consensus,
            'dimensions': dimensions,
            'grade': self.get_grade(total_consensus)
        }
    
    def map_category_to_dimension(self, category):
        """映射质疑类型到维度"""
        mapping = {
            ChallengeCategory.REQUIREMENT: 'requirement',
            ChallengeCategory.TECHNICAL: 'technical',
            ChallengeCategory.ARCHITECTURE: 'technical',
            ChallengeCategory.COST: 'cost',
            ChallengeCategory.SCHEDULE: 'schedule',
            ChallengeCategory.RISK: 'risk',
            ChallengeCategory.RESOURCE: 'schedule',
            ChallengeCategory.COMPLIANCE: 'risk',
        }
        return mapping.get(category, 'technical')
    
    def get_grade(self, consensus):
        """获取共识等级"""
        if consensus >= 85:
            return 'A - 高度共识'
        elif consensus >= 70:
            return 'B - 基本共识'
        elif consensus >= 50:
            return 'C - 部分分歧'
        elif consensus >= 30:
            return 'D - 较大分歧'
        else:
            return 'E - 严重分歧'
```
