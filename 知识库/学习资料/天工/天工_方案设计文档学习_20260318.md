// 签名验证拦截器
@Component
public class SignatureInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) {
        String appId = request.getHeader("appId");
        String timestamp = request.getHeader("timestamp");
        String nonce = request.getHeader("nonce");
        String sign = request.getHeader("sign");
        
        // 1. 验证时间戳有效性
        if (System.currentTimeMillis() / 1000 - Long.parseLong(timestamp) > 300) {
            throw new SignException("请求已过期");
        }
        
        // 2. 验证nonce防重放
        if (nonceCache.exists(nonce)) {
            throw new SignException("重复请求");
        }
        
        // 3. HMAC-Sha256签名验证
        String expectedSign = HmacSha256.sign(appId, body, timestamp, nonce, secretKey);
        if (!expectedSign.equals(sign)) {
            throw new SignException("签名验证失败");
        }
        
        return true;
    }
}

### 4.3 商机属性脱敏实现

#### 文档定义的脱敏规则：
| 序号 | 字段分类 | 商机属性 | 脱敏规则 |
|-----|---------|---------|---------|
| 1 | 物理空间 | 客户地址 | 仅展示前5个字，其他** |
| 2 | 客户信息 | 客户名称 | 仅展示第一个字，后面*** |
| 3 | 客户信息 | 客户联系方式 | 中间四位脱密**** |
| 4 | 客户信息 | 身份证号码 | 展示前2位和结尾2位，中间*** |

#### 脱敏服务实现：
```java
@Service
public class DataMaskService {
    
    /**
     * 商机属性脱敏
     * 文档要求：
     * 1. 登录时将用户角色脱敏规则存入缓存
     * 2. 查询商机时根据缓存中的规则脱敏
     */
    public void maskOppAttrs(List<OppAttr> attrs, Long roleId) {
        // 1. 从缓存获取角色脱敏配置
        List<AttrMaskRule> rules = cacheService.getMaskRules(roleId);
        
        for (OppAttr attr : attrs) {
            AttrMaskRule rule = rules.stream()
                .filter(r -> r.getAttrCode().equals(attr.getAttrCode()))
                .findFirst().orElse(null);
            
            if (rule != null) {
                attr.setAttrValue(maskValue(attr.getAttrValue(), rule.getMaskType()));
            }
        }
    }
    
    private String maskValue(String value, String maskType) {
        switch (maskType) {
            case "ADDRESS":
                // 地址：仅展示前5个字，其他**
                return value.length() > 5 ? value.substring(0, 5) + "**" : value;
            case "NAME":
                // 姓名：仅展示第一个字
                return value.substring(0, 1) + "***";
            case "PHONE":
                // 手机号：中间四位脱敏
                return value.replaceAll("(\\d{3})\\d{4}(\\d{4})", "$1****$2");
            case "ID_CARD":
                // 身份证：前2位+后2位
                return value.substring(0, 2) + "******" + value.substring(value.length() - 2);
            default:
                return value;
        }
    }
}
```

### 6.2 数据清洗流程实现

#### 文档定义的清洗环节：
1. 红黑名单过滤
2. 合规性拦截（敏感词）
3. 规则清洗和数据修正
4. 异常处理

```java
@Service
public class OppDataCleanService {
    
    /**
     * 数据清洗主流程
     */
    public CleanResult clean(OppData oppData) {
        CleanResult result = new CleanResult();
        
        try {
            // 1. 红黑名单过滤
            if (!checkBlackWhiteList(oppData, result)) {
                return result;
            }
            
            // 2. 合规性拦截
            if (!checkCompliance(oppData, result)) {
                return result;
            }
            
            // 3. 规则清洗
            cleanByRules(oppData, result);
            
            // 4. 数据修正
            fixData(oppData, result);
            
            result.setSuccess(true);
            
        } catch (Exception e) {
            result.setSuccess(false);
            result.setErrorMsg(e.getMessage());
        }
        
        return result;
    }
    
    /**
     * 敏感词检测 - AC自动机算法
     */
    private boolean checkCompliance(OppData oppData, CleanResult result) {
        for (OppAttr attr : oppData.getAttrs()) {
            if (containsChineseOrEnglish(attr.getAttrValue())) {
                List<String> sensitiveWords = acAutomaton.match(attr.getAttrValue());
                if (!sensitiveWords.isEmpty()) {
                    result.addProblem(attr.getAttrCode(), "存在敏感词：" + sensitiveWords);
                    return false;
                }
            }
        }
        return true;
    }
    
    /**
     * 数据修正示例 - 日期格式
     */
    private void fixData(OppData oppData, CleanResult result) {
        for (OppAttr attr : oppData.getAttrs()) {
            if ("DATE".equals(attr.getInputType()) && attr.isAbnormal()) {
                String fixed = fixDateFormat(attr.getAttrValue());
                if (fixed != null) {
                    result.addChange(attr.getAttrCode(), attr.getAttrValue(), fixed);
                    attr.setAttrValue(fixed);
                    attr.setAttrSource("1100"); // 修正来源
                }
            }
        }
    }
    
    /**
     * 日期格式修正
     * 文档规则：20251020 → 2025-10-20
     */
    private String fixDateFormat(String value) {
        // 20251020 → 2025-10-20
        if (value.matches("^\\d{8}$")) {
            return value.substring(0, 4) + "-" + value.substring(4, 6) + "-" + value.substring(6, 8);
        }
        // 2025/10/20 → 2025-10-20
        if (value.matches("^\\d{4}/\\d{2}/\\d{2}$")) {
            return value.replace("/", "-");
        }
        return null;
    }
}
```

        LocalDate archiveDate = LocalDate.now().minusYears(2);
        
        // 归档商机主表
        oppMapper.archive(archiveDate);
        // 归档商机属性表
        oppAttrMapper.archive(archiveDate);
    }
}
