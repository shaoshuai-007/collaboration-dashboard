## 一、Redis缓存策略

### 1.1 缓存穿透、缓存击穿、缓存雪崩

#### 缓存穿透（Cache Penetration）
**定义**：查询不存在的数据，缓存和数据库都没有，导致每次请求都打到数据库。

**解决方案**：
```python
# 方案1：布隆过滤器
from pybloom_live import BloomFilter

bf = BloomFilter(capacity=1000000, error_rate=0.001)

def get_user(user_id):
    if user_id not in bf:
        return None  # 直接返回，不查数据库
    
    # 正常缓存查询逻辑
    cache_key = f"user:{user_id}"
    # ...

# 方案2：空值缓存
def get_user(user_id):
    cache_key = f"user:{user_id}"
    data = redis.get(cache_key)
    
    if data == "NULL":  # 空值标记
        return None
    
    if data:
        return json.loads(data)
    
    # 查询数据库
    user = db.query_user(user_id)
    if user:
        redis.setex(cache_key, 3600, json.dumps(user))
    else:
        redis.setex(cache_key, 60, "NULL")  # 缓存空值，短过期
    
    return user
```

**应用场景**：
- 电信系统：查询不存在的用户套餐
- 电商系统：查询不存在的商品

#### 缓存击穿（Cache Breakdown）
**定义**：热点key过期，大量并发请求同时打到数据库。

**解决方案**：
```python
import threading
import time

def get_hot_data(key):
    cache_key = f"hot:{key}"
    data = redis.get(cache_key)
    
    if data:
        return json.loads(data)
    
    # 加锁，防止并发
    lock_key = f"lock:{key}"
    lock = redis.set(lock_key, "1", nx=True, ex=10)
    
    if lock:
        try:
            # 双重检查
            data = redis.get(cache_key)
            if data:
                return json.loads(data)
            
            # 查询数据库
            result = db.query(key)
            redis.setex(cache_key, 3600, json.dumps(result))
            return result
        finally:
            redis.delete(lock_key)
    else:
        # 等待重试
        time.sleep(0.1)
        return get_hot_data(key)
```

**应用场景**：
- 电信套餐详情页
- 限时秒杀商品

#### 缓存雪崩（Cache Avalanche）
**定义**：大量key同时过期，导致数据库压力骤增。

**解决方案**：
```python
import random

def set_cache_with_jitter(key, value, base_ttl=3600):
    """设置缓存，添加随机过期时间"""
    jitter = random.randint(0, 300)  # 0-5分钟随机偏移
    ttl = base_ttl + jitter
    redis.setex(key, ttl, value)

# 批量设置时，确保过期时间分散
def batch_set_cache(items):
    for i, (key, value) in enumerate(items):
        # 基础过期时间 + 随机偏移
        ttl = 3600 + random.randint(0, 600)
        redis.setex(key, ttl, value)
```

**应用场景**：
- 批量导入数据后预热
- 系统重启后缓存重建

```python
def warm_up_cache():
    """系统启动时预热热点数据"""

```

#### 缓存更新策略
```python
# Cache-Aside Pattern（旁路缓存）
def get_data(key):
    # 先查缓存
    data = redis.get(key)
    if data:
        return json.loads(data)
    
    # 缓存未命中，查数据库
    data = db.query(key)
    if data:
        redis.setex(key, 3600, json.dumps(data))
    return data

def update_data(key, value):
    # 先更新数据库
    db.update(key, value)
    # 再删除缓存（而非更新）
    redis.delete(key)
```

### 1.3 分布式锁实现

```python
import uuid
import time

class DistributedLock:
    def __init__(self, redis_client):
        self.redis = redis_client
    
    def acquire(self, lock_name, timeout=10):
        """获取分布式锁"""
        identifier = str(uuid.uuid4())
        lock_key = f"lock:{lock_name}"
        
        # SET NX EX 原子操作
        if self.redis.set(lock_key, identifier, nx=True, ex=timeout):
            return identifier
        return None
    
    def release(self, lock_name, identifier):
        """释放分布式锁"""
        lock_key = f"lock:{lock_name}"
        
        # Lua脚本确保原子性
        script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        self.redis.eval(script, 1, lock_key, identifier)

# 使用示例
lock = DistributedLock(redis)

def process_order(order_id):
    lock_id = lock.acquire(f"order:{order_id}")
    if not lock_id:
        raise Exception("获取锁失败，请稍后重试")
    
    try:
        # 处理订单逻辑
        pass
    finally:
        lock.release(f"order:{order_id}", lock_id)
```

```python
import pika

# 生产者
def send_message(queue_name, message):
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()
    channel.queue_declare(queue=queue_name)
    channel.basic_publish(
        exchange='',
        routing_key=queue_name,
        body=json.dumps(message)
    )
    connection.close()

# 消费者
def consume_message(queue_name, callback):
    connection = pika.BlockingConnection(
        pika.ConnectionParameters('localhost')
    )
    channel = connection.channel()
    channel.queue_declare(queue=queue_name)
    channel.basic_consume(
        queue=queue_name,
        on_message_callback=callback,
        auto_ack=True
    )
    channel.start_consuming()
```

### 2.3 事件驱动架构

```python
# 事件总线
class EventBus:
    def __init__(self):
        self.handlers = {}
    
    def subscribe(self, event_type, handler):
        if event_type not in self.handlers:
            self.handlers[event_type] = []
        self.handlers[event_type].append(handler)
    
    def publish(self, event_type, data):
        handlers = self.handlers.get(event_type, [])
        for handler in handlers:
            handler(data)

# 使用示例
event_bus = EventBus()

# 订阅事件
def on_user_registered(user_data):
    send_welcome_email(user_data)
    create_user_profile(user_data)

event_bus.subscribe('user_registered', on_user_registered)

# 发布事件
event_bus.publish('user_registered', {'user_id': 123, 'email': 'user@example.com'})
```

### 3.1 SQL优化

#### 索引优化
```sql
-- 创建索引
CREATE INDEX idx_user_phone ON users(phone);
CREATE INDEX idx_order_user_date ON orders(user_id, created_at);

-- 复合索引原则：最左前缀
-- 索引 (a, b, c) 支持：a, a+b, a+b+c
-- 不支持：b, c, b+c

-- 查看执行计划
EXPLAIN SELECT * FROM users WHERE phone = '13800138000';
```

#### 慢查询优化
```sql
-- 避免 SELECT *
SELECT id, name, phone FROM users WHERE status = 1;

-- 避免 LIKE '%xxx%' 全模糊匹配
SELECT * FROM users WHERE name LIKE '张%';  -- 使用索引

-- 避免在 WHERE 子句中对字段进行函数操作
-- 错误
SELECT * FROM orders WHERE DATE(created_at) = '2024-01-01';
-- 正确
SELECT * FROM orders WHERE created_at >= '2024-01-01' AND created_at < '2024-01-02';
```

### 3.2 代码优化

#### 算法优化
```python
# 错误：O(n²) 查找
def find_duplicates(list1, list2):
    result = []
    for item in list1:
        if item in list2:  # O(n)
            result.append(item)
    return result  # 总复杂度 O(n*m)

# 正确：O(n) 查找
def find_duplicates(list1, list2):
    set2 = set(list2)  # O(m)
    return [item for item in list1 if item in set2]  # O(n)
    # 总复杂度 O(n + m)
```

#### 数据结构选择
