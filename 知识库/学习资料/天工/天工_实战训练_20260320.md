#### 数据库设计

```sql
-- 用户表
CREATE TABLE t_user (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键ID',
    user_name       VARCHAR(50) NOT NULL COMMENT '用户名',
    phone           VARCHAR(20) NOT NULL COMMENT '手机号',
    package_id      INT COMMENT '当前套餐ID',
    value_score     INT DEFAULT 0 COMMENT '价值分数',
    status          TINYINT DEFAULT 1 COMMENT '状态：1正常 0停用',
    create_time     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    UNIQUE KEY uk_phone (phone),
    KEY idx_package_id (package_id),
    KEY idx_value_score (value_score)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

#### API设计

```
接口列表：
```

#### 核心代码实现

```python
from flask import Flask, jsonify, request
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql://user:pass@localhost/db'
db = SQLAlchemy(app)

class User(db.Model):
    """用户模型"""
    __tablename__ = 't_user'

    id = db.Column(db.BigInteger, primary_key=True)
    user_name = db.Column(db.String(50), nullable=False)
    phone = db.Column(db.String(20), unique=True, nullable=False)
    package_id = db.Column(db.Integer)
    value_score = db.Column(db.Integer, default=0)
    status = db.Column(db.SmallInteger, default=1)
    create_time = db.Column(db.DateTime, default=datetime.utcnow)
    update_time = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    def to_dict(self):
        """转换为字典"""
        return {
            'id': self.id,
            'user_name': self.user_name,
            'phone': self.phone,
            'package_id': self.package_id,
            'value_score': self.value_score,
            'status': self.status,
            'create_time': self.create_time.isoformat() if self.create_time else None,
            'update_time': self.update_time.isoformat() if self.update_time else None
        }

@app.route('/api/v1/users', methods=['GET'])
def get_users():
    """
    获取用户列表

    Args:
        page: 页码（默认1）
        size: 每页数量（默认20）

    Returns:
        用户列表
    """
    page = request.args.get('page', 1, type=int)
    size = request.args.get('size', 20, type=int)

    pagination = User.query.filter_by(status=1).paginate(page=page, per_page=size)
    users = pagination.items

    return jsonify({
        'code': 200,
        'message': 'success',
        'data': {
            'total': pagination.total,
            'page': page,
            'size': size,
            'items': [u.to_dict() for u in users]
        }
    })

@app.route('/api/v1/users/<int:user_id>', methods=['GET'])
def get_user(user_id):
    """
    获取单个用户

    Args:
        user_id: 用户ID

    Returns:
        用户详情
    """
    user = User.query.get(user_id)
    if not user:
        return jsonify({'code': 404, 'message': '用户不存在'}), 404

    return jsonify({
        'code': 200,
        'message': 'success',
        'data': user.to_dict()
    })

@app.route('/api/v1/users', methods=['POST'])
def create_user():
    """
    创建用户

    Args:
        user_name: 用户名
        phone: 手机号
        package_id: 套餐ID（可选）

    Returns:
        创建的用户信息
    """
    data = request.get_json()

    # 参数校验
    if not data.get('user_name') or not data.get('phone'):
        return jsonify({'code': 400, 'message': '参数不完整'}), 400

    # 检查手机号是否已存在
    if User.query.filter_by(phone=data['phone']).first():
        return jsonify({'code': 400, 'message': '手机号已存在'}), 400

    # 创建用户
    user = User(
        user_name=data['user_name'],
        phone=data['phone'],
        package_id=data.get('package_id')
    )
    db.session.add(user)
    db.session.commit()

    return jsonify({
        'code': 200,
        'message': '创建成功',
        'data': user.to_dict()
    }), 201

@app.route('/api/v1/users/<int:user_id>', methods=['PUT'])
def update_user(user_id):
    """
    更新用户

    Args:
        user_id: 用户ID
        user_name: 用户名（可选）
        package_id: 套餐ID（可选）
        value_score: 价值分数（可选）

    Returns:
        更新后的用户信息
    """
    user = User.query.get(user_id)
    if not user:
        return jsonify({'code': 404, 'message': '用户不存在'}), 404

    data = request.get_json()

    # 更新字段
    if 'user_name' in data:
        user.user_name = data['user_name']
    if 'package_id' in data:
        user.package_id = data['package_id']
    if 'value_score' in data:
        user.value_score = data['value_score']

    db.session.commit()

    return jsonify({
        'code': 200,
        'message': '更新成功',
        'data': user.to_dict()
    })

@app.route('/api/v1/users/<int:user_id>', methods=['DELETE'])
def delete_user(user_id):
    """
    删除用户（逻辑删除）

    Args:
        user_id: 用户ID

    Returns:
        删除结果
    """
    user = User.query.get(user_id)
    if not user:
        return jsonify({'code': 404, 'message': '用户不存在'}), 404

    user.status = 0  # 逻辑删除
    db.session.commit()

    return jsonify({
        'code': 200,
        'message': '删除成功'
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
```

```python
        'user_name': '测试用户',

def test_delete_user(client):
    """测试删除用户"""
    # 创建用户
    response = client.post('/api/v1/users', json={
        'user_name': '测试用户',
        'phone': '13800138003'
    })
    user_id = response.get_json()['data']['id']

    # 删除用户
    response = client.delete(f'/api/v1/users/{user_id}')
    assert response.status_code == 200

    # 验证已删除
    response = client.get(f'/api/v1/users/{user_id}')
```
