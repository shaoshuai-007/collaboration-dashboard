```python
class BillingEngine:
    """电信计费引擎"""
    
    def calculate_usage_fee(self, user_id: str, usage_data: dict) -> Decimal:
        """
        计算使用费用
        
        Args:
            user_id: 用户ID
            usage_data: 使用数据（通话时长/流量/短信数）
        
        Returns:
            计算后的费用金额
        """
        # 1. 获取用户套餐信息
        package = self.get_user_package(user_id)
        
        # 2. 检查套餐余量
        remaining = self.check_package_remaining(user_id, package)
        
        # 3. 计算超出部分
        overage = max(usage_data['amount'] - remaining, 0)
        
        # 4. 应用阶梯定价
        if overage > 0:
            fee = self.apply_tiered_pricing(overage, package.pricing_tiers)
        else:
            fee = Decimal('0.00')
        
        return fee
    
    def apply_tiered_pricing(self, amount: float, tiers: list) -> Decimal:
        """阶梯定价计算"""
        total_fee = Decimal('0.00')
        remaining = amount
        
        for tier in sorted(tiers, key=lambda x: x['start']):
            tier_amount = min(remaining, tier['end'] - tier['start'])
            total_fee += Decimal(str(tier_amount * tier['price']))
            remaining -= tier_amount
            
            if remaining <= 0:
                break
        
        return total_fee
```

```python
class PackageRecommender:
    """套餐推荐引擎"""
    
    def recommend(self, user_profile: dict, top_n: int = 3) -> list:
        """
        智能推荐套餐
        
        Args:
            user_profile: 用户画像数据
            top_n: 返回前N个推荐
        
        Returns:
            推荐套餐列表
        """
        candidates = []
        
        # 1. 基于用户画像筛选候选套餐
        for package in self.get_available_packages():
            score = self.calculate_match_score(user_profile, package)
            if score > 0.6:  # 匹配度阈值
                candidates.append({
                    'package': package,
                    'score': score,
                    'savings': self.calculate_savings(user_profile, package)
                })
        
        # 2. 按综合评分排序
        candidates.sort(key=lambda x: x['score'], reverse=True)
        
        # 3. 返回Top N
        return candidates[:top_n]
    
    def calculate_match_score(self, profile: dict, package: dict) -> float:
        """计算套餐匹配分数"""
        score = 0.0
        
        # 语音匹配度 (权重0.3)
        voice_match = self._match_voice(profile['avg_voice'], package['voice'])
        score += voice_match * 0.3
        
        # 流量匹配度 (权重0.4)
        data_match = self._match_data(profile['avg_data'], package['data'])
        score += data_match * 0.4
        
        # 价格匹配度 (权重0.3)
        price_match = self._match_price(profile['budget'], package['monthly_fee'])
        score += price_match * 0.3
        
        return score
```

```python
"""
用户服务模块

提供用户管理相关业务逻辑
"""

from typing import Optional, List
from datetime import datetime
from dataclasses import dataclass


@dataclass
class UserInfo:
    """用户信息数据类"""
    user_id: str
    user_name: str
    phone_number: str
    create_time: datetime


class UserService:
    """用户业务服务类
    
    提供用户的增删改查操作
    
    Attributes:
        db: 数据库连接
        cache: 缓存客户端
    """
    
    # 类常量
    MAX_QUERY_LIMIT = 100
    DEFAULT_PAGE_SIZE = 20
    
    def __init__(self, db_connection, cache_client):
        """初始化用户服务
        
        Args:
            db_connection: 数据库连接对象
            cache_client: Redis缓存客户端
        """
        self._db = db_connection
        self._cache = cache_client
    
    def get_user_by_id(self, user_id: str) -> Optional[UserInfo]:
        """根据ID获取用户信息
        
        Args:
            user_id: 用户唯一标识
        
        Returns:
            用户信息对象，不存在则返回None
        
        Raises:
            ValueError: 当user_id为空时
        """
        if not user_id:
            raise ValueError("user_id不能为空")
        
        # 先查缓存
        cache_key = f"user:info:{user_id}"
        cached = self._cache.get(cache_key)
        if cached:
            return UserInfo(**cached)
        
        # 查数据库
        sql = "SELECT * FROM user_base_info WHERE user_id = %s"
        result = self._db.execute_query(sql, (user_id,))
        
        if result:
            user = UserInfo(**result[0])
            # 写入缓存
            self._cache.set(cache_key, result[0], ex=3600)
            return user
        
        return None
    
    def list_users(
        self, 
        page: int = 1, 
        page_size: int = None,
        status: int = None
    ) -> List[UserInfo]:
        """查询用户列表
        
        Args:
            page: 页码，从1开始
            page_size: 每页数量，默认20
            status: 用户状态过滤
        
        Returns:
            用户信息列表
        """
        page_size = page_size or self.DEFAULT_PAGE_SIZE
        offset = (page - 1) * page_size
        
        conditions = ["1=1"]
        params = []
        
        if status is not None:
            conditions.append("status = %s")
            params.append(status)
        
        sql = f"""
            SELECT * FROM user_base_info 
            WHERE {' AND '.join(conditions)}
            LIMIT %s OFFSET %s
        """
        params.extend([page_size, offset])
        
        results = self._db.execute_query(sql, tuple(params))
        return [UserInfo(**r) for r in results]
```

### 1.3.3 Redis缓存策略

#### 缓存模式
```python
import redis
import json
from typing import Optional, Any
from functools import wraps

class CacheManager:
    """Redis缓存管理器"""
    
    def __init__(self, host: str, port: int, db: int = 0):
        self.client = redis.Redis(
            host=host,
            port=port,
            db=db,
            decode_responses=True
        )
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        value = self.client.get(key)
        if value:
            return json.loads(value)
        return None
    
    def set(self, key: str, value: Any, ex: int = 3600):
        """设置缓存"""
        self.client.set(key, json.dumps(value), ex=ex)
    
    def delete(self, key: str):
        """删除缓存"""
        self.client.delete(key)
    
    def cache_aside(self, key: str, ttl: int = 3600):
        """Cache-Aside模式装饰器"""
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                # 先查缓存
                cached = self.get(key)
                if cached is not None:
                    return cached
                
                # 缓存未命中，执行函数
                result = func(*args, **kwargs)
                
                # 写入缓存
                if result is not None:
                    self.set(key, result, ex=ttl)
                
                return result
            return wrapper
        return decorator


# 使用示例
cache = CacheManager('localhost', 6379)

def get_user_info(user_id: str):
    """获取用户信息（带缓存）"""
    # 数据库查询逻辑
```

#### 缓存穿透/击穿/雪崩解决方案
```python
class CacheProtection:
    """缓存保护机制"""
    
    def __init__(self, cache_client):
        self.cache = cache_client
        self.bloom_filter = BloomFilter()
    
    def get_with_protection(self, key: str, db_query, ttl: int = 3600):
        """带保护的缓存获取"""
        
        # 1. 布隆过滤器防止穿透
        if not self.bloom_filter.might_contain(key):
            return None  # 一定不存在
        
        # 2. 查缓存
        value = self.cache.get(key)
        if value is not None:
            return value
        
        # 3. 分布式锁防止击穿
        lock_key = f"lock:{key}"
        if self.cache.client.setnx(lock_key, 1):
            self.cache.client.expire(lock_key, 10)
            try:
                # 查数据库
                value = db_query()
                if value:
                    # 缓存结果
                    self.cache.set(key, value, ex=ttl)
                else:
                    # 空值缓存防止穿透
                    self.cache.set(key, "NULL", ex=60)
                return value
            finally:
                self.cache.delete(lock_key)
        else:
            # 等待并重试
            time.sleep(0.1)
            return self.get_with_protection(key, db_query, ttl)
```

#### JWT工具类
```python
from datetime import datetime, timedelta
from typing import Optional
import jwt
from fastapi import HTTPException, Security
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

class JWTAuth:
    """JWT认证工具"""
    
    def __init__(self, secret_key: str, algorithm: str = "HS256"):
        self.secret_key = secret_key
        self.algorithm = algorithm
        self.security = HTTPBearer()
    
    def create_token(
        self, 
        user_id: str, 
        expires_hours: int = 24,
        **extra_claims
    ) -> str:
        """生成JWT Token"""
        payload = {
            "sub": user_id,
            "iat": datetime.utcnow(),
            "exp": datetime.utcnow() + timedelta(hours=expires_hours),
            **extra_claims
        }
        return jwt.encode(payload, self.secret_key, algorithm=self.algorithm)
    
    def verify_token(
        self, 
        credentials: HTTPAuthorizationCredentials = Security(HTTPBearer())
    ) -> dict:
        """验证JWT Token"""
        try:
            token = credentials.credentials
            payload = jwt.decode(
                token, 
                self.secret_key, 
                algorithms=[self.algorithm]
            )
            return payload
        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=401, detail="Token已过期")
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=401, detail="无效的Token")
    
    def refresh_token(self, old_token: str) -> str:
        """刷新Token"""
        payload = self.verify_token(old_token)
        user_id = payload.get("sub")
        return self.create_token(user_id)
```

### 3.3.2 推荐服务实现

```python
# app/services/recommend_service.py
from typing import List, Optional, Dict
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)


@dataclass
class PackageInfo:
    """套餐信息"""
    package_id: str
    package_name: str
    package_type: int
    monthly_fee: float
    voice_amount: int
    data_amount: int
    sms_amount: int
    voice_overage: float
    data_overage: float


@dataclass
class UserProfile:
    """用户画像"""
    user_id: str
    avg_voice: float  # 月均语音分钟
    avg_data: float   # 月均流量MB
    avg_sms: float    # 月均短信
    current_package_id: str
    monthly_spend: float  # 月均消费


class RecommendService:
    """智能配案推荐服务"""
    
    def __init__(self, db, cache):
        self.db = db
        self.cache = cache
    
    def recommend_packages(
        self, 
        user_id: str, 
        preferences: Optional[dict] = None
    ) -> Optional[dict]:
        """
        生成套餐推荐
        
        Args:
            user_id: 用户ID
            preferences: 用户偏好设置
        
        Returns:
            推荐结果
        """
        # 1. 获取用户画像
        profile = self._build_user_profile(user_id)
        if not profile:
            logger.warning(f"用户画像构建失败: {user_id}")
            return None
        
        # 2. 获取当前套餐
        current_package = self._get_current_package(profile.current_package_id)
        
        # 3. 获取候选套餐
        candidates = self._get_candidate_packages(profile, preferences)
        
        # 4. 计算推荐分数
        recommendations = []
        for pkg in candidates:
            score = self._calculate_match_score(profile, pkg, preferences)
            savings = self._calculate_savings(profile, pkg)
            reasons = self._generate_reasons(profile, pkg, savings)
            
            recommendations.append({
                "package_id": pkg.package_id,
                "package_name": pkg.package_name,
                "monthly_fee": pkg.monthly_fee,
                "voice_amount": pkg.voice_amount,
                "data_amount": pkg.data_amount,
                "match_score": round(score, 4),
                "savings_amount": round(savings, 2),
                "reasons": reasons
            })
        
        # 5. 按分数排序，取Top 3
        recommendations.sort(key=lambda x: x["match_score"], reverse=True)
        top_recommendations = recommendations[:3]
        
        # 6. 保存推荐记录
        self._save_recommend_log(user_id, top_recommendations)
        
        return {
            "user_id": user_id,
            "current_package": {
                "package_id": current_package.package_id,
                "package_name": current_package.package_name,
                "monthly_fee": current_package.monthly_fee
            },
            "recommendations": top_recommendations,
            "analysis": {
                "avg_voice": profile.avg_voice,
                "avg_data": profile.avg_data,
                "monthly_spend": profile.monthly_spend
            }
        }
    
    def _build_user_profile(self, user_id: str) -> Optional[UserProfile]:
        """构建用户画像"""
        sql = """
            SELECT 
                user_id,
                package_id,
                COALESCE((
                    SELECT AVG(voice_usage) 
                    FROM usage_log 
                    WHERE user_id = %s 
                    AND usage_type = 'voice'
                ), 0) as avg_voice,
                COALESCE((
                    SELECT AVG(data_usage) 
                    FROM usage_log 
                    WHERE user_id = %s 
                    AND usage_type = 'data'
                ), 0) as avg_data,
                COALESCE((
                    SELECT AVG(monthly_fee)
                    FROM billing_record
                    WHERE user_id = %s
                ), 0) as monthly_spend
            FROM user_base_info
            WHERE user_id = %s
        """
        result = self.db.execute_query(sql, (user_id, user_id, user_id, user_id))
        
        if not result:
            return None
        
        row = result[0]
        return UserProfile(
            user_id=row['user_id'],
            avg_voice=row['avg_voice'] or 0,
            avg_data=row['avg_data'] or 0,
            avg_sms=0,
            current_package_id=row['package_id'],
            monthly_spend=row['monthly_spend'] or 0
        )
    
    def _calculate_match_score(
        self, 
        profile: UserProfile, 
        package: PackageInfo,
        preferences: Optional[dict] = None
    ) -> float:
        """计算匹配分数"""
        score = 0.0
        
        # 语音匹配度 (权重0.3)
        if package.voice_amount > 0:
            voice_ratio = min(profile.avg_voice / package.voice_amount, 1.0)
            voice_score = voice_ratio if voice_ratio > 0.5 else voice_ratio * 0.8
        else:
            voice_score = 0.5
        score += voice_score * 0.3
        
        # 流量匹配度 (权重0.4)
        if package.data_amount > 0:
            data_ratio = min(profile.avg_data / package.data_amount, 1.0)
            data_score = data_ratio if data_ratio > 0.5 else data_ratio * 0.8
        else:
            data_score = 0.5
        score += data_score * 0.4
        
        # 价格匹配度 (权重0.3)
        if preferences and 'budget' in preferences:
            budget = preferences['budget']
            if package.monthly_fee <= budget:
                price_score = 1.0 - (package.monthly_fee / budget) * 0.5
            else:
                price_score = max(0, 1.0 - (package.monthly_fee - budget) / budget)
        else:
            price_score = 0.5
        score += price_score * 0.3
        
        return score
    
    def _calculate_savings(self, profile: UserProfile, package: PackageInfo) -> float:
        """计算节省金额"""
        current_fee = profile.monthly_spend
        new_fee = package.monthly_fee
        
        # 计算超出费用
        overage_fee = 0
        if profile.avg_voice > package.voice_amount:
            overage_fee += (profile.avg_voice - package.voice_amount) * package.voice_overage
        if profile.avg_data > package.data_amount:
            overage_fee += (profile.avg_data - package.data_amount) * package.data_overage
        
        new_total = new_fee + overage_fee
        return current_fee - new_total
    
    def _generate_reasons(
        self, 
        profile: UserProfile, 
        package: PackageInfo, 
        savings: float
    ) -> List[str]:
        """生成推荐理由"""
        reasons = []
        
        # 流量评估
        if package.data_amount >= profile.avg_data * 1.2:
            reasons.append(f"流量充足({package.data_amount}MB)，满足您的使用需求")
        elif package.data_amount >= profile.avg_data:
            reasons.append(f"流量适中({package.data_amount}MB)，基本满足使用")
        
        # 语音评估
        if package.voice_amount >= profile.avg_voice * 1.2:
            reasons.append(f"语音分钟数充足({package.voice_amount}分钟)")
        
        # 节省评估
        if savings > 0:
            reasons.append(f"月费降低，预计每月节省{savings:.0f}元")
        elif savings == 0:
            reasons.append("月费基本持平")
        
        return reasons
    
    def _save_recommend_log(self, user_id: str, recommendations: List[dict]):
        """保存推荐记录"""
        for rec in recommendations:
            sql = """
                INSERT INTO recommend_log 
                (user_id, recommend_package_id, match_score, savings_amount)
                VALUES (%s, %s, %s, %s)
            """
            self.db.execute_insert(sql, (
                user_id,
                rec['package_id'],
                rec['match_score'],
                rec['savings_amount']
            ))
```
